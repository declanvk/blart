8/// This struct contains the running state needed to calculate an
9/// [`InnerNodeWidthDistribution`] when fed a sorted list of keys.
10#[derive(Debug)]
11struct CalculateInnerNodeDist {
12    last_key_bytes: Vec<u8>,
13    is_first: bool,
14    dist: InnerNodeWidthDistribution,
15    prefix_stack: Vec<PrefixFrame>,
16}
17
18#[derive(Debug)]
19struct PrefixFrame {
20    prefix_length: usize,
21    width: usize,
22}
23
24impl<K> Extend<K> for CalculateInnerNodeDist
25where
26    K: AsBytes,
27{
28    fn extend<T: IntoIterator<Item = K>>(&mut self, iter: T) {
29        self.ingest(iter);
30    }
31}

46impl CalculateInnerNodeDist {
47    fn new() -> Self {
48        Self {
49            last_key_bytes: Vec::new(),
50            is_first: true,
51            dist: InnerNodeWidthDistribution::default(),
52            prefix_stack: Vec::new(),
53        }
54    }
55
56    /// Consume the given keys and add the results to the running inner node
57    /// width distribution.
58    ///
59    /// This function assumes that the list of keys is sorted by their byte
60    /// representation (as returned by [`AsBytes::as_bytes`]), otherwise it will
61    /// return incorrect results. Also, the keys should be globally sorted over
62    /// all invocations of [`ingest`][Self::ingest].
63    ///
64    /// # Panic
65    ///
66    /// This function also assumes that there are no key is a prefix of any
67    /// other key in the list, other it may panic.
68    fn ingest<K: AsBytes>(&mut self, keys: impl IntoIterator<Item = K>) {
69        let mut keys = keys.into_iter();
70        while let Some(key) = keys.next() {
71            let key_bytes = key.as_bytes();
72
73            let prefix_length = self.calculate_longest_common_prefix(key_bytes);
74            self.update_last_key_bytes(key_bytes);
75
76            loop {
77                let Some(last_frame) = self.prefix_stack.last_mut() else {
78                    // TODO: Figure out why we push a width of 0 for the top of the stack?????
79                    self.prefix_stack.push(PrefixFrame {
80                        prefix_length,
81                        width: 0,
82                    });
83                    break;
84                };
85
86                if last_frame.prefix_length > prefix_length {
87                    // Then we've finished processing all children of the top frame
88                    let prefix_frame = self.prefix_stack.pop().unwrap();
89                    Self::write_frame_into_dist(&mut self.dist, prefix_frame);
90                    continue;
91                } else if last_frame.prefix_length < prefix_length {
92                    self.prefix_stack.push(PrefixFrame {
93                        prefix_length,
94                        width: 1,
95                    });
96                    break;
97                } else {
98                    last_frame.width += 1;
99                    break;
100                }
101            }
102        }
103    }
104
105    /// Consume the running state to produce the final inner node width
106    /// distribution.
107    fn finish(mut self) -> InnerNodeWidthDistribution {
108        for prefix_frame in self.prefix_stack {
109            Self::write_frame_into_dist(&mut self.dist, prefix_frame);
110        }
111
112        self.dist
113    }
114
115    fn calculate_longest_common_prefix(&mut self, new_key_bytes: &[u8]) -> usize {
116        let last_key_bytes = self.last_key_bytes.as_slice();
117
118        if !self.is_first {
119            // Since the `keys` iterator is assumed to be ordered by byte representation,
120            // the shorter key will always go before the longer key, so we only need to
121            // check if `last_key_bytes` is the prefix.
122            assert!(
123                !new_key_bytes.starts_with(&last_key_bytes),
124                "no prefixed keys allowed. {last_key_bytes:?} {new_key_bytes:?}"
125            );
126        } else {
127            self.is_first = false;
128        }
129
130        // longest common prefix
131        last_key_bytes
132            .iter()
133            .zip(new_key_bytes)
134            .take_while(|(a, b)| **a == **b)
135            .count()
136    }
137
138    fn update_last_key_bytes(&mut self, new_key_bytes: &[u8]) {
139        self.last_key_bytes.clear();
140        self.last_key_bytes.extend_from_slice(new_key_bytes);
141    }
142
143    fn write_frame_into_dist(
144        dist: &mut InnerNodeWidthDistribution,
145        PrefixFrame { width, .. }: PrefixFrame,
146    ) {
147        // TODO: Explain why we add 1 here!
148        let width = width + 1;
149        if width > 1 {
150            dist.add(
151                InnerNodeWidth::try_from(width).expect("width should fall within expected bounds"),
152                1,
153            );
154        }
155    }
156}

158/// Given a sorted iterator of keys, compute the distribution of inner node
159/// widths.
160pub fn analyze_sorted_key_iter<K>(keys: impl Iterator<Item = K>) -> InnerNodeWidthDistribution
161where
162    K: AsBytes,
163{
164    let mut calculator = CalculateInnerNodeDist::new();
165
166    calculator.ingest(keys);
167

171/// This struct represents a distribution of inner node widths gathered
172/// from a set of keys.
173#[derive(Clone, PartialEq, Eq)]
174pub struct InnerNodeWidthDistribution {
175    /// This field store an array for each of the possible node width, from `[2,
176    /// 256]`.
177    ///
178    /// It is not possible to have a node width of 0 or 1 (or greater than 256).
179    /// Because of that, we store node width 2 at the 0th index, node width 3 at
180    /// the 1st index, etc etc, and node width 256 at the 254th index.
181    counts: [usize; 255],
182}
183
184impl Default for InnerNodeWidthDistribution {
185    fn default() -> Self {
186        Self { counts: [0; 255] }
187    }
188}
189
190impl fmt::Debug for InnerNodeWidthDistribution {
191    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
192        let mut builder = f.debug_struct("InnerNodeWidthDistribution");
193        for (width, count) in self.iter().filter(|(_, count)| *count != 0) {
194            builder.field(&format!("{width}"), &count);
195        }
196        builder.finish()
197    }
198}

200impl InnerNodeWidthDistribution {
201    /// For the [`InnerNodeWidth`] increase the corresponding count by `count`.
202    #[inline]
203    pub fn add(&mut self, width: InnerNodeWidth, count: usize) {
204        self.counts[width.get() - 2] += count;
205    }
206
207    /// Retrieve the count for the given [`InnerNodeWidth`]
208    #[inline]
209    pub fn get(&self, width: InnerNodeWidth) -> usize {
210        self.counts[width.get() - 2]
211    }
212
213    /// Return an iterator over all [`InnerNodeWidth`]s with non-zero count.
214    pub fn iter(&self) -> impl Iterator<Item = (InnerNodeWidth, usize)> + use<'_> {
215        self.counts
216            .iter()
217            .enumerate()
218            .map(|(idx, count)| (InnerNodeWidth::try_from(idx + 2).unwrap(), *count))
219            .filter(|(_, count)| *count != 0)
220    }
221
222    /// Find the optimal assignment of `num_inner_nodes - 1` inner nodes using
223    /// this inner node width distribution as a factor in the cost.
224    ///
225    /// The optimization currently only takes into account the size of the node.
226    /// See [`InnerNodeKind`] for details. A consequence of this is that the
227    /// optimization results will only ever return inner node assignments of
228    /// [`InnerNodeKind::KeyChildCompressed`] because it is always smaller than
229    /// [`InnerNodeKind::ChildOnlyCompressed`].
230    ///
231    /// A future version of this function could decide to take into account the
232    /// relative lookup or update performance of the two kinds and return a
233    /// different result.
234    pub fn find_optimal_assignment(&self, num_inner_nodes: usize) -> InnerNodeWidthAssignment {
235        assert!(num_inner_nodes > 1);
236        assert!(num_inner_nodes <= 256 - 2);
237        // 1 inner node width is always assigned to 256 width
238        let num_inner_nodes_to_assign = num_inner_nodes - 1;
239
240        // memo[k][w], where k ranges over the node widths we need to assign and w
241        // ranges over the possible values for the width [2, 255].
242        const POSSIBLE_WIDTHS: usize = 256 - 2;
243        let mut memo =
244            vec![f64::MAX; POSSIBLE_WIDTHS * num_inner_nodes_to_assign * InnerNodeKind::ALL.len()];
245        let mut backpointers =
246            vec![
247                None::<(InnerNodeWidth, InnerNodeKind)>;
248                POSSIBLE_WIDTHS * num_inner_nodes_to_assign * InnerNodeKind::ALL.len()
249            ];
250
251        fn to_index(inner_node: usize, width: usize, kind: usize) -> usize {
252            const NODE_STRIDE: usize = POSSIBLE_WIDTHS * InnerNodeKind::ALL.len();
253            const WIDTH_STRIDE: usize = InnerNodeKind::ALL.len();
254            inner_node * NODE_STRIDE + width * WIDTH_STRIDE + kind
255        }
256
257        // Base case - since we know there is at least 1 inner node to assign (first
258        // assert)
259        for width in (InnerNodeWidth::MIN.get()..InnerNodeWidth::MAX.get())
260            .map(|w| InnerNodeWidth::new(w).unwrap())
261        {
262            for kind in InnerNodeKind::ALL {
263                memo[to_index(0, width.get() - InnerNodeWidth::MIN.get(), kind as usize)] =
264                    cost_between(None, width, kind, self);
265            }
266        }
267
268        // Recurrence - use the previous width calculated to find the optimal new width
269        for inner_node in 1..num_inner_nodes_to_assign {
270            for width in (InnerNodeWidth::MIN.get()..InnerNodeWidth::MAX.get())
271                .map(|w| InnerNodeWidth::new(w).unwrap())
272            {
273                for kind in InnerNodeKind::ALL {
274                    let mut minimum_cost = f64::MAX;
275                    let mut previous_width_with_min_cost = None;
276                    for prev_width in (InnerNodeWidth::MIN.get()..width.get())
277                        .map(|w| InnerNodeWidth::new(w).unwrap())
278                    {
279                        let cost = cost_between(Some(prev_width), width, kind, self)
280                            + memo[to_index(
281                                inner_node - 1,
282                                prev_width.get() - InnerNodeWidth::MIN.get(),
283                                kind as usize,
284                            )];
285                        if cost < minimum_cost {
286                            minimum_cost = cost;
287                            previous_width_with_min_cost = Some((prev_width, kind));
288                        }
289                    }
290                    let index = to_index(
291                        inner_node,
292                        width.get() - InnerNodeWidth::MIN.get(),
293                        kind as usize,
294                    );
295                    memo[index] = minimum_cost;
296                    backpointers[index] = previous_width_with_min_cost;
297                }
298            }
299        }
300
301        // Compute the minimum cost when considering a final node256 at the end
302        let mut minimum_cost = f64::MAX;
303        let mut minimum_cost_prev_width = None;
304        for kind in InnerNodeKind::ALL {
305            for prev_width in (InnerNodeWidth::MIN.get()..InnerNodeWidth::MAX.get())
306                .map(|w| InnerNodeWidth::new(w).unwrap())
307            {
308                let index = to_index(
309                    num_inner_nodes_to_assign - 1,
310                    prev_width.get() - InnerNodeWidth::MIN.get(),
311                    kind as usize,
312                );
313                let cost =
314                    cost_between(Some(prev_width), InnerNodeWidth::MAX, kind, self) + memo[index];
315                if cost < minimum_cost {
316                    minimum_cost = cost;
317                    minimum_cost_prev_width = Some((prev_width, kind));
318                }
319            }
320        }
321
322        let mut assignment = Vec::with_capacity(num_inner_nodes_to_assign);
323        let mut node_kinds = Vec::with_capacity(num_inner_nodes_to_assign);
324
325        // Follow the back pointers to find the optimal assignment
326        let mut current_minimum_cost_width = minimum_cost_prev_width;
327        let mut next_inner_node = num_inner_nodes_to_assign - 1;
328        while let Some((prev_width, prev_kind)) = current_minimum_cost_width {
329            let index = to_index(
330                next_inner_node,
331                prev_width.get() - InnerNodeWidth::MIN.get(),
332                prev_kind as usize,
333            );
334            assignment.push(prev_width);
335            node_kinds.push(prev_kind);
336            current_minimum_cost_width = backpointers[index];
337            next_inner_node = next_inner_node.saturating_sub(1);
338        }
339        assignment.reverse();
340        node_kinds.reverse();
341
342        InnerNodeWidthAssignment::new(assignment.into_boxed_slice(), node_kinds.into_boxed_slice())
343            .unwrap()
344    }
345}

347/// This enum represents the different kinds of inner nodes.
348#[derive(Debug, Clone, Copy, PartialEq, Eq)]
349pub enum InnerNodeKind {
350    /// This kind matches the `Node4` or `Node16` from the original ART paper,
351    /// or a [`blart::raw::InnerNodeSorted`].
352    ///
353    /// In an inner node with width `w`, nodes of this kind have `w` values of
354    /// `u8` representing the child key bytes and `w` values of roughly `*const
355    /// ()` representing the pointers to children.
356    KeyChildCompressed = 0,
357    /// This kind matches the `Node48` from the original ART paper, or a
358    /// [`blart::raw::InnerNode48`].
359    ///
360    /// In an inner node with width `w`, nodes of this kind have `256` values of
361    /// `u8` representing the child key bytes and `w` values of roughly `*const
362    /// ()` representing the pointers to children.
363    ChildOnlyCompressed = 1,
364}
365
366impl InnerNodeKind {
367    const ALL: [Self; 2] = [Self::KeyChildCompressed, Self::ChildOnlyCompressed];
368}

386/// This type represents an inner node width in the range `[2, 256]`.
387#[derive(Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
388pub struct InnerNodeWidth(NonZeroU8);
389
390impl InnerNodeWidth {
391    /// The maximum inner node width of 256.
392    pub const MAX: Self = Self(NonZeroU8::MAX);
393    /// The minimum inner node width of 2.
394    pub const MIN: Self = Self(NonZeroU8::MIN);
395
396    /// Return a new [`InnerNodeWidth`] if the value is within `[2, 256]`,
397    /// otherwise return `None`.
398    pub const fn new(width: usize) -> Option<Self> {
399        if width < 2 || width > 256 {
400            return None;
401        }
402
403        Some(Self(NonZeroU8::new((width - 1) as u8).unwrap()))
404    }
405
406    /// Return the width value as a `usize`.
407    pub const fn get(self) -> usize {
408        // TODO: Use `usize::from` when trait methods are stable in const contexts
409        (self.0.get() as usize) + 1
410    }
411
412    /// Return the next larger [`InnerNodeWidth`], or `None` if `self` is
413    /// [`Self::MAX`].
414    pub fn increment(self) -> Option<Self> {
415        self.0.checked_add(1).map(Self)
416    }
417}

419impl fmt::Display for InnerNodeWidth {
420    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
421        write!(f, "n{}", self.get())
422    }
423}

426    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
427        f.debug_tuple("InnerNodeWidth").field(&self.get()).finish()
428    }

467/// This struct represents a set of inner nodes of specific widths and kinds.
468///
469/// The assignment aspect is because typically this set is smaller than the
470/// number of inner node widths in the original [`InnerNodeWidthDistribution`],
471/// so the assignment will have some cost.
472///
473/// See [`InnerNodeKind`] for the different types of inner nodes.
474#[derive(Debug, Clone, PartialEq, Eq)]
475pub struct InnerNodeWidthAssignment {
476    widths: Box<[InnerNodeWidth]>,
477    kinds: Box<[InnerNodeKind]>,
478}
479
480impl Default for InnerNodeWidthAssignment {
481    fn default() -> Self {
482        Self {
483            widths: Box::from([
484                InnerNodeWidth::new(4).unwrap(),
485                InnerNodeWidth::new(16).unwrap(),
486                InnerNodeWidth::new(48).unwrap(),
487            ]),
488            kinds: Box::from([
489                InnerNodeKind::KeyChildCompressed,
490                InnerNodeKind::KeyChildCompressed,
491                InnerNodeKind::ChildOnlyCompressed,
492            ]),
493        }
494    }
495}
496
497impl InnerNodeWidthAssignment {
498    /// Create a new assignment with the given widths and kinds.
499    ///
500    /// The widths are checked to make sure that:
501    ///   1. [`InnerNodeWidth::MAX`] is not present
502    ///   2. The list of widths is sorted
503    ///
504    /// Otherwise, this functions returns `None`.
505    pub fn new(
506        widths: impl Into<Box<[InnerNodeWidth]>>,
507        kinds: impl Into<Box<[InnerNodeKind]>>,
508    ) -> Option<Self> {
509        let widths = widths.into();
510        let kinds = kinds.into();
511        let contains_256_width = widths
512            .as_ref()
513            .iter()
514            .any(|width| *width == InnerNodeWidth::MAX);
515        let is_sorted = widths.as_ref().is_sorted();
516        if contains_256_width || !is_sorted {
517            return None;
518        }
519
520        Some(Self { widths, kinds })
521    }
522
523    fn iter(&self) -> impl Iterator<Item = (InnerNodeWidth, InnerNodeKind)> + use<'_> {
524        self.widths.iter().copied().zip(self.kinds.iter().copied())
525    }
526
527    /// Calculate the cost for this assignment given the inner node width
528    /// distribution.
529    pub fn cost_for(&self, dist: &InnerNodeWidthDistribution) -> f64 {
530        let staggered_iter = iter::once(None).chain(self.iter().map(|(w, _)| Some(w)));
531
532        let mut cost = 0.0;
533        for (prev_width, (curr_width, curr_kind)) in staggered_iter.zip(self.iter()) {
534            cost += cost_between(prev_width, curr_width, curr_kind, dist);
535        }
536
537        cost
538    }
539}

541/// Calculate the cost for using an inner node of `new_width` preceded by a
542/// inner node of width `prev_width` with the given `kind` calculated against
543/// the given distribution.
544///
545/// This function allows for incrementally calculating the assignment cost by
546/// summing the cost between successive pairs of widths in the assignment.
547fn cost_between(
548    prev_width: Option<InnerNodeWidth>,
549    new_width: InnerNodeWidth,
550    kind: InnerNodeKind,
551    dist: &InnerNodeWidthDistribution,
552) -> f64 {
553    let mut current_width = match prev_width {
554        // This is the typical case, where we're dealing with a previous width ranging from [2,
555        // 256]
556        Some(width) => {
557            match width.increment() {
558                Some(width) => width,
559                // In this case we overflow the width value, so the range of possible costs
560                // would have been empty. Returning 0 to represent 0 cost
561                None => return 0.0,
562            }
563        },
564        // This case indicates that there is no previous width and we're starting with the very
565        // first width
566        None => InnerNodeWidth::MIN,
567    };
568
569    let bytes_per_node = match kind {
570        InnerNodeKind::KeyChildCompressed => {
571            // for a node of width w, this kind has size (w * 1) + (w *
572            // mem::size_of::<* const ()>())
573            new_width.get() + (new_width.get() * mem::size_of::<*const ()>())
574        },
575        InnerNodeKind::ChildOnlyCompressed => {
576            // for a node of width w, this kind has a size of (256 * 1) + (w
577            // * mem::size_of::<* const ()>())
578            256 + new_width.get() * mem::size_of::<*const ()>()
579        },
580    };
581    let mut num_nodes = 0;
582    while current_width <= new_width {
583        let count = dist.get(current_width);
584        num_nodes += count;
585
586        current_width = match current_width.increment() {
587            Some(w) => w,
588            None => break,
589        };
590    }
591
592    // This calculation of the cost only considers the space usage of the nodes.
593    // Under this criteria, the `InnerNodeKind::KeyChildCompressed` will always have
594    // better space usage than `InnerNodeKind::ChildOnlyCompressed`. A future
595    // extension would be to calculate some weighted score that also consider
596    // relative operation speed, since lookup (and maybe update) should be faster
597    // for the `ChildOnlyCompressed` node kind.
598    let total_cost = (bytes_per_node * num_nodes) as f64;
599
600    total_cost
601}
