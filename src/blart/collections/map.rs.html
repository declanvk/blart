278    /// Consume the tree, returning a raw pointer to the root node and the
279    /// allocator of the tree.
280    ///
281    /// If the results is `None`, this means the tree is empty.
282    ///
283    /// # Examples
284    ///
285    /// ```rust
286    /// use blart::TreeMap;
287    ///
288    /// let mut map = TreeMap::<Box<[u8]>, char>::new();
289    ///
290    /// map.try_insert(Box::new([1, 2, 3]), 'a').unwrap();
291    /// assert_eq!(map.len(), 1);
292    ///
293    /// let (root, alloc) = TreeMap::into_raw_with_allocator(map);
294    /// assert!(root.is_some());
295    ///
296    /// // SAFETY: The root pointer came directly from the `into_raw` result.
297    /// let _map = unsafe { TreeMap::from_raw_in(root, alloc) }.unwrap();
298    /// ```
299    pub fn into_raw_with_allocator(tree: Self) -> (Option<OpaqueNodePtr<K, V, PREFIX_LEN>>, A) {
300        // We need this `ManuallyDrop` so that the `TreeMap::drop` is not called.
301        // Since the `root` field is `Copy`, it can be moved out of the tree without
302        // inhibiting `Drop`
303        let tree = ManuallyDrop::new(tree);
304        // SAFETY: Since we're reading from an `&A` that was coerced to a `*const A` we
305        // know that the pointer is valid for reads, properly aligned, and properly
306        // initialized.
307        //
308        // Also this is safe from a double-free since we're using `ManuallyDrop` to
309        // inhibit the first copy of `A` (in the `tree` value) from doing anything.
310        let alloc = unsafe { ptr::read(&tree.alloc) };
311        let root = tree.state.as_ref().map(|state| state.root);
312
313        (root, alloc)
314    }

378    pub unsafe fn from_raw_in(
379        root: Option<OpaqueNodePtr<K, V, PREFIX_LEN>>,
380        alloc: A,
381    ) -> Result<Self, MalformedTreeError<K, V, PREFIX_LEN>>
382    where
383        K: AsBytes,
384    {
385        match root {
386            Some(root) => {
387                // SAFETY: The safety doc of this function guarantees the uniqueness of the
388                // `root` pointer, which means we won't have any other mutations
389                let stats = unsafe { WellFormedChecker::check_tree(root)? };
390
391                let (min_leaf, max_leaf) =
392                    // SAFETY: The safety doc of this function guarantees the uniqueness of the
393                // `root` pointer, which means we won't have any other mutations
394                    unsafe { (minimum_unchecked(root), maximum_unchecked(root)) };
395
396                Ok(Self {
397                    state: Some(NonEmptyTree {
398                        root,
399                        min_leaf,
400                        max_leaf,
401                    }),
402                    num_entries: stats.num_leaf,
403                    alloc,
404                })
405            },
406            None => Ok(Self::with_prefix_len_in(alloc)),

430    /// Returns the key-value pair corresponding to the supplied key.
431    ///
432    /// # Examples
433    ///
434    /// ```rust
435    /// use blart::TreeMap;
436    ///
437    /// let mut map = TreeMap::<Box<[u8]>, char>::new();
438    ///
439    /// map.try_insert(Box::new([1, 2, 3]), 'a').unwrap();
440    /// assert_eq!(map.get_key_value([1, 2, 3].as_ref()).unwrap(), (&Box::from([1, 2, 3]), &'a'));
441    /// ```
442    pub fn get_key_value<Q>(&self, key: &Q) -> Option<(&K, &V)>
443    where
444        K: Borrow<Q> + AsBytes,
445        Q: AsBytes + ?Sized,
446    {
447        if let Some(state) = &self.state {
448            // SAFETY: Since we have an immutable reference to the `TreeMap` object, that
449            // means there can only exist other immutable references aside from this one,
450            // and no mutable references. That means that no mutating operations can occur
451            // on the root node or any child of the root node.
452            let search_result = unsafe { search_unchecked(state.root, key.as_bytes())? };
453
454            // SAFETY: The lifetime chosen the value reference is bounded by the lifetime of
455            // the immutable reference to the `TreeMap`. The memory of the value will not be
456            // mutated since it is only owned by the `TreeMap` and there can only be other
457            // immutable references at this time (no mutable references to the `TreeMap`).
458            let (key, value) = unsafe { search_result.as_key_value_ref() };
459            Some((key, value))
460        } else {
461            None
462        }
463    }

482    /// on the tree can happen.

495    /// assert_eq!(*map[[1, 2, 3].as_ref()], 'a');
496    ///
497    /// *value = 'b';
498    /// drop(value);

500    pub fn get_mut<Q>(&mut self, key: &Q) -> Option<&mut V>
501    where
502        K: Borrow<Q> + AsBytes,
503        Q: AsBytes + ?Sized,
504    {
505        if let Some(state) = &self.state {
506            // SAFETY: Since we have a mutable reference to the `TreeMap` object, that
507            // means there cannot exist any other reference (mutable or immutable) to the
508            // same `TreeMap`. Which means that no other mutating operations could be
509            // happening during the `search_unchecked` call.
510            let search_result = unsafe { search_unchecked(state.root, key.as_bytes())? };
511
512            // SAFETY: The lifetime chosen the value reference is bounded by the lifetime of
513            // the mutable reference to the `TreeMap`. The value pointed to by the returned
514            // mutable reference will not be accessed (read or written) through any other
515            // pointer because of the existing mutable reference on the `TreeMap`.
516            let leaf_node_ref = unsafe { search_result.as_mut() };
517
518            Some(leaf_node_ref.value_mut())
519        } else {
520            None
521        }
522    }

746    fn apply_insert_point(
747        &mut self,
748        insert_point: InsertPoint<K, V, PREFIX_LEN>,
749        key: K,
750        value: V,
751    ) -> InsertResult<'_, K, V, PREFIX_LEN>
752    where
753        K: AsBytes,
754    {
755        // SAFETY:
756        //  - This call is safe because we have a mutable reference on the tree, so no
757        //    other operation can be concurrent with this one.
758        //  - The same allocator is used for all inserts and deletes
759        let insert_result = unsafe { insert_point.apply(key, value, &self.alloc) };
760
761        match &mut self.state {
762            Some(state) => {
763                state.root = insert_result.new_root;
764
765                {
766                    // SAFETY: This call is safe because we have a mutable reference on the tree and
767                    // the returned reference is bounded to this block, not returned
768                    let new_leaf = unsafe { insert_result.leaf_node_ptr.as_ref() };
769                    if new_leaf.previous.is_none() {
770                        state.min_leaf = insert_result.leaf_node_ptr;
771                    }
772
773                    if new_leaf.next.is_none() {
774                        state.max_leaf = insert_result.leaf_node_ptr;
775                    }
776                }
777            },
778            None => {
779                self.state = Some(NonEmptyTree {
780                    root: insert_result.new_root,
781                    min_leaf: insert_result.leaf_node_ptr,
782                    max_leaf: insert_result.leaf_node_ptr,
783                })
784            },
785        }
786
787        if insert_result.existing_leaf.is_none() {
788            // this was a strict add, not a replace. If there was an existing leaf we are
789            // removing and adding a leaf, so the number of entries stays the same
790            self.num_entries += 1;
791        }
792
793        insert_result
794    }

796    fn apply_delete_point(
797        &mut self,
798        delete_point: DeletePoint<K, V, PREFIX_LEN>,
799    ) -> DeleteResult<K, V, PREFIX_LEN> {
800        // SAFETY:
801        // - The root is sure to not be `None`, since the we somehow got a
802        //   `DeletePoint`. So the caller must have checked this. Also, since we have a
803        //   mutable reference to the tree, no other read or write operation can be
804        //   happening concurrently.
805        // - `self.alloc` is the same allocator which is used for all inserts and
806        //   deletes on this trie
807        let delete_result =
808            unsafe { delete_point.apply(self.state.as_ref().unwrap_unchecked().root, &self.alloc) };
809
810        match &mut self.state {
811            Some(state) => {
812                if let Some(new_root) = delete_result.new_root {
813                    state.root = new_root;
814
815                    if delete_result.deleted_leaf.previous.is_none() {
816                        state.min_leaf = delete_result.deleted_leaf.next.expect(
817                            "this should be Some since this is the non-singleton delete case",
818                        );
819                    }
820
821                    if delete_result.deleted_leaf.next.is_none() {
822                        state.max_leaf = delete_result.deleted_leaf.previous.expect(
823                            "this should be Some since this is the non-singleton delete case",
824                        );
825                    }
826                } else {
827                    self.state = None;
828                }
829            },
830            None => unreachable!("a successful deletion requires a non-empty tree"),
831        }
832
833        self.num_entries -= 1;
834
835        delete_result
836    }

907                let insert_point = search_for_insert_point(state.root, key.as_bytes())?;

909            };
910            Ok(insert_result.existing_leaf.map(|leaf| leaf.into_entry().1))

1458    /// Tries to get the given key’s corresponding entry in the map for in-place
1459    /// manipulation.
1460    pub fn try_entry(&mut self, key: K) -> Result<Entry<'_, K, V, PREFIX_LEN, A>, InsertPrefixError>
1461    where
1462        K: AsBytes,
1463    {
1464        let entry = match &self.state {
1465            Some(state) => {
1466                // SAFETY: Since we have a mutable reference to the `TreeMap`, we are guaranteed
1467                // that there are no other references (mutable or immutable) to this same
1468                // object. Meaning that our access to the root node is unique and there are no
1469                // other accesses to any node in the tree.
1470                let insert_point = unsafe { search_for_insert_point(state.root, key.as_bytes())? };
1471                match insert_point.insert_type {
1472                    Exact { leaf_node_ptr } => Entry::Occupied(OccupiedEntry {
1473                        map: self,
1474                        leaf_node_ptr,
1475                        grandparent_ptr_and_parent_key_byte: insert_point
1476                            .grandparent_ptr_and_parent_key_byte,
1477                        parent_ptr_and_child_key_byte: insert_point.parent_ptr_and_child_key_byte,
1478                    }),
1479                    _ => Entry::Vacant(VacantEntry {
1480                        key,
1481                        insert_point: Some(insert_point),
1482                        map: self,
1483                    }),
1484                }
1485            },
1486            None => Entry::Vacant(VacantEntry {
1487                key,
1488                insert_point: None,
1489                map: self,
1490            }),
1491        };
1492        Ok(entry)
1493    }

2332                block: UnsafeCell::new(MaybeUninit::uninit()),
2333
2334                // These three will be fixed up after allocating the allocator
2335                start: Cell::new(NonNull::dangling()),
2336                end: Cell::new(NonNull::dangling()),

2338

2348                NonNull::new(unsafe { alloc_start.as_ptr().offset(N.try_into().unwrap()) })
2349                    .unwrap(),

2358        #[inline]
2359        fn allocate(&self, layout: Layout) -> Result<NonNull<[u8]>, AllocError> {
2360            self.alloc_count.set(self.alloc_count.get() + 1);
2361
2362            let size = layout.size();
2363            let align = layout.align();
2364
2365            debug_assert!(align > 0);
2366            debug_assert!(align.is_power_of_two());
2367
2368            let ptr = nonnull_addr(self.ptr.get());
2369
2370            let new_ptr = ptr.get().checked_sub(size).unwrap();
2371
2372            // Round down to the requested alignment.
2373            let new_ptr = NonZeroUsize::new(new_ptr & !(align - 1)).unwrap();
2374
2375            let start = nonnull_addr(self.start.get());
2376            if new_ptr < start {
2377                // Didn't have enough capacity!
2378                return Err(AllocError);
2379            }
2380
2381            self.ptr.set(nonnull_with_addr(self.ptr.get(), new_ptr));
2382            Ok(NonNull::slice_from_raw_parts(self.ptr.get(), size))
2383        }
2384
2385        #[inline]
2386        unsafe fn deallocate(&self, _ptr: NonNull<u8>, _layout: Layout) {
2387            self.dealloc_count.set(self.dealloc_count.get() + 1);
2388        }
