21/// This struct contains details of where and why the search stopped in an inner
22/// node.
23pub(crate) struct InnerNodeSearchResult<K, V, const PREFIX_LEN: usize> {
24    /// This is a pointer to the inner node where search stopped.
25    pub node_ptr: OpaqueNodePtr<K, V, PREFIX_LEN>,
26
27    /// The number of bytes of the key that were consumed, not including the
28    /// byte that was used to look for a child node.
29    pub current_depth: usize,
30
31    /// This is a comparison from the full prefix of the inner node to the
32    /// corresponding segment of the search key.
33    pub node_prefix_comparison_to_search_key_segment: Ordering,
34
35    /// This field indicates why the search stopped in the inner node.
36    pub reason: InnerNodeSearchResultReason,
37}
38
39impl<K, V, const PREFIX_LEN: usize> fmt::Debug for InnerNodeSearchResult<K, V, PREFIX_LEN> {
40    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
41        f.debug_struct("InnerNodeSearchResult")
42            .field("node_ptr", &self.node_ptr)
43            .field("current_depth", &self.current_depth)
44            .field(
45                "node_prefix_comparison_to_search_key_segment",
46                &self.node_prefix_comparison_to_search_key_segment,
47            )
48            .field("reason", &self.reason)
49            .finish()
50    }
51}

53/// These are search termination reasons specific to inner nodes.
54#[derive(Debug)]
55pub(crate) enum InnerNodeSearchResultReason {
56    /// This variant means the search terminated in a mismatched prefix of an
57    /// inner node OR the prefix matched, but there were insufficient key bytes
58    /// to lookup a child of the inner node.
59    PrefixMismatchOrInsufficientBytes,
60    /// This variant means the search terminated in an inner node, when there
61    /// was no corresponding child for a key byte.
62    MissingChild,
63}
64
65/// This enum provides the different cases of search result when looking for a
66/// starting/ending point for iteration.
67pub(crate) enum TerminatingNodeSearchResult<K, V, const PREFIX_LEN: usize> {
68    /// This variant means the search terminated at a leaf node.
69    Leaf {
70        /// The leaf node pointer where search terminated.
71        leaf_ptr: NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>,
72        /// The ordering of the leaf key bytes relative to the search bytes.
73        leaf_key_comparison_to_search_key: Ordering,
74    },
75    /// This variant means the search terminated at an inner node.
76    InnerNode(InnerNodeSearchResult<K, V, PREFIX_LEN>),
77}

79impl<K, V, const PREFIX_LEN: usize> fmt::Debug for TerminatingNodeSearchResult<K, V, PREFIX_LEN> {
80    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
81        match self {
82            Self::Leaf {
83                leaf_ptr,
84                leaf_key_comparison_to_search_key,
85            } => f
86                .debug_struct("Leaf")
87                .field("leaf_ptr", leaf_ptr)
88                .field(
89                    "leaf_key_comparison_to_search_key",
90                    leaf_key_comparison_to_search_key,
91                )
92                .finish(),
93            Self::InnerNode(arg0) => f.debug_tuple("InnerNode").field(arg0).finish(),
94        }
95    }

98/// Find the node (inner or leaf) that is identified by the given bytestring.
99///
100/// # Safety
101///
102/// This function cannot be called concurrently with any mutating operation
103/// on `root` or any child node of `root`. This function will arbitrarily
104/// read to any child in the given tree.
105pub(crate) unsafe fn find_terminating_node<K: AsBytes, V, const PREFIX_LEN: usize>(
106    root: OpaqueNodePtr<K, V, PREFIX_LEN>,
107    key_bytes: &[u8],
108) -> TerminatingNodeSearchResult<K, V, PREFIX_LEN> {
109    /// This function will search some key bytes at a specific depth in the
110    /// given inner node, returning different results depending on if the search
111    /// terminated or needs to continue down the trie.
112    ///
113    /// If this function returns [`ControlFlow::Continue`] then the search
114    /// continues down the trie, otherwise it terminates at this inner node.
115    ///
116    /// # Safety
117    ///
118    /// Callers must guarantee that there is no concurrent mutation of the given
119    /// inner node for the duration of this function.
120    unsafe fn check_prefix_lookup_child<K, V, N, const PREFIX_LEN: usize>(
121        inner_ptr: NodePtr<PREFIX_LEN, N>,
122        key_bytes: &[u8],
123        current_depth: &mut usize,
124        prefix_match_behavior: &mut PrefixMatchBehavior,
125    ) -> ControlFlow<InnerNodeSearchResult<K, V, PREFIX_LEN>, OpaqueNodePtr<K, V, PREFIX_LEN>>
126    where
127        N: InnerNode<PREFIX_LEN, Key = K, Value = V>,
128        K: AsBytes,
129    {
130        // SAFETY: The lifetime produced from this is bounded to this scope and does not
131        // escape. Further, no other code mutates the node referenced, which is further
132        // enforced the "no concurrent reads or writes" requirement on the
133        // `check_prefix_lookup_child` function.
134        let inner_node = unsafe { inner_ptr.as_ref() };
135        let match_prefix =
136            prefix_match_behavior.match_prefix(inner_node, &key_bytes[*current_depth..]);
137        match match_prefix {
138            Err(_) => {
139                let (full_prefix, _) = inner_node.read_full_prefix(*current_depth);
140                let upper_bound = (*current_depth + full_prefix.len()).min(key_bytes.len());
141                let key_segment = &key_bytes[(*current_depth)..upper_bound];
142
143                let node_prefix_comparison_to_search_key_segment = full_prefix.cmp(key_segment);
144
145                ControlFlow::Break(InnerNodeSearchResult {
146                    node_ptr: inner_ptr.to_opaque(),
147                    current_depth: *current_depth,
148                    reason: InnerNodeSearchResultReason::PrefixMismatchOrInsufficientBytes,
149                    node_prefix_comparison_to_search_key_segment,
150                })
151            },
152            Ok(AttemptOptimisticPrefixMatch { matched_bytes, .. }) => {
153                // Since the prefix matched, advance the depth by the size of the prefix
154                *current_depth += matched_bytes;
155
156                let next_key_fragment = if *current_depth < key_bytes.len() {
157                    key_bytes[*current_depth]
158                } else {
159                    // the key has insufficient bytes, it is a prefix of an existing key. Thus, the
160                    // key must not exist in the tree by the requirements of the insert function
161                    // (any key in the tree must not be the prefix of any other key in the tree)
162                    return ControlFlow::Break(InnerNodeSearchResult {
163                        node_ptr: inner_ptr.to_opaque(),
164                        current_depth: *current_depth,
165                        reason: InnerNodeSearchResultReason::PrefixMismatchOrInsufficientBytes,
166                        node_prefix_comparison_to_search_key_segment: Ordering::Equal,
167                    });
168                };
169
170                let child_lookup = inner_node.lookup_child(next_key_fragment);
171
172                match child_lookup {
173                    Some(child_ptr) => {
174                        // Since the prefix matched and it found a child, advance the depth by 1
175                        // more key byte
176                        *current_depth += 1;
177                        ControlFlow::Continue(child_ptr)
178                    },
179                    None => ControlFlow::Break(InnerNodeSearchResult {
180                        node_ptr: inner_ptr.to_opaque(),
181                        current_depth: *current_depth,
182                        reason: InnerNodeSearchResultReason::MissingChild,
183                        node_prefix_comparison_to_search_key_segment: Ordering::Equal,
184                    }),
185                }
186            },
187        }
188    }
189
190    let mut current_node = root;
191    let mut current_depth = 0;
192    let mut prefix_match_behavior = PrefixMatchBehavior::default();
193
194    loop {
195        let next_node = match current_node.to_node_ptr() {
196            ConcreteNodePtr::Node4(inner_ptr) => unsafe {
197                // SAFETY: The safety requirement is covered by the safety documentation on the
198                // containing function
199                check_prefix_lookup_child(
200                    inner_ptr,
201                    key_bytes,
202                    &mut current_depth,
203                    &mut prefix_match_behavior,
204                )
205            },
206            ConcreteNodePtr::Node16(inner_ptr) => unsafe {
207                // SAFETY: The safety requirement is covered by the safety documentation on the
208                // containing function
209                check_prefix_lookup_child(
210                    inner_ptr,
211                    key_bytes,
212                    &mut current_depth,
213                    &mut prefix_match_behavior,
214                )
215            },
216            ConcreteNodePtr::Node48(inner_ptr) => unsafe {
217                // SAFETY: The safety requirement is covered by the safety documentation on the
218                // containing function
219                check_prefix_lookup_child(
220                    inner_ptr,
221                    key_bytes,
222                    &mut current_depth,
223                    &mut prefix_match_behavior,
224                )
225            },
226            ConcreteNodePtr::Node256(inner_ptr) => unsafe {
227                // SAFETY: The safety requirement is covered by the safety documentation on the
228                // containing function
229                check_prefix_lookup_child(
230                    inner_ptr,
231                    key_bytes,
232                    &mut current_depth,
233                    &mut prefix_match_behavior,
234                )
235            },
236            ConcreteNodePtr::LeafNode(leaf_ptr) => {
237                // SAFETY: The shared reference is bounded to this block and there are no
238                // concurrent modifications, by the safety conditions of this function.
239                let leaf_node = unsafe { leaf_ptr.as_ref() };
240
241                let leaf_key_comparison_to_search_key =
242                    prefix_match_behavior.compare_leaf_key(leaf_node, key_bytes, current_depth);
243
244                return TerminatingNodeSearchResult::Leaf {
245                    leaf_ptr,
246                    leaf_key_comparison_to_search_key,
247                };
248            },
249        };
250
251        match next_node {
252            ControlFlow::Continue(next_node) => {
253                current_node = next_node;
254            },
255            ControlFlow::Break(reason) => {
256                return TerminatingNodeSearchResult::InnerNode(reason);
257            },
258        }
259    }
260}

262/// Panic if the given range bounds
263#[track_caller]
264pub(crate) fn validate_range_bounds(start_bound: &Bound<&[u8]>, end_bound: &Bound<&[u8]>) {
265    match (*start_bound, *end_bound) {
266        (Bound::Excluded(s), Bound::Excluded(e)) if s == e => {
267            panic!("range start and end are equal and excluded: ({s:?})")
268        },
269        (Bound::Included(s) | Bound::Excluded(s), Bound::Included(e) | Bound::Excluded(e))
270            if s > e =>
271        {
272            panic!("range start ({s:?}) is greater than range end ({e:?})")
273        },
274        _ => {},
275    }
276}

278type KeyByteAndChild<K, V, const PREFIX_LEN: usize> = Option<(u8, OpaqueNodePtr<K, V, PREFIX_LEN>)>;

280/// This function searches a trie for the smallest/largest leaf node that is
281/// greater/less than the given bound.
282///
283/// If `is_start` is true, its looking for the smallest leaf node that is
284/// greater than the given bound. If `is_start` is false, we're looking for the
285/// greatest leaf node that is smaller than the given bound.
286///
287/// # Safety
288///
289/// Callers must guarantee that there is no concurrent mutation of the given
290/// trie for the duration of this function.
291pub(crate) unsafe fn find_leaf_pointer_for_bound<K: AsBytes, V, const PREFIX_LEN: usize>(
292    tree: &NonEmptyTree<K, V, PREFIX_LEN>,
293    bound: Bound<&[u8]>,
294    is_start: bool,
295) -> Option<NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>> {
296    Some(match bound {
297        Bound::Included(key_bytes) | Bound::Excluded(key_bytes) => {
298            // SAFETY: Covered by function safety doc
299            let terminating_node = unsafe { find_terminating_node(tree.root, key_bytes) };
300
301            match terminating_node {
302                TerminatingNodeSearchResult::Leaf {
303                    leaf_ptr,
304                    leaf_key_comparison_to_search_key,
305                } => match leaf_key_comparison_to_search_key {
306                    Ordering::Less => {
307                        if is_start {
308                            // SAFETY: Covered by function safety doc
309                            let leaf = unsafe { leaf_ptr.as_ref() };
310                            leaf.next?
311                        } else {
312                            leaf_ptr
313                        }
314                    },
315                    Ordering::Equal => {
316                        if matches!(bound, Bound::Excluded(_)) {
317                            // SAFETY: Covered by function safety doc
318                            let leaf = unsafe { leaf_ptr.as_ref() };
319                            if is_start { leaf.next } else { leaf.previous }?
320                        } else {
321                            leaf_ptr
322                        }
323                    },
324                    Ordering::Greater => {
325                        if is_start {
326                            leaf_ptr
327                        } else {
328                            // SAFETY: Covered by function safety doc
329                            let leaf = unsafe { leaf_ptr.as_ref() };
330                            leaf.previous?
331                        }
332                    },
333                },
334                TerminatingNodeSearchResult::InnerNode(InnerNodeSearchResult {
335                    node_ptr,
336                    current_depth,
337                    reason,
338                    node_prefix_comparison_to_search_key_segment,
339                }) => match reason {
340                    InnerNodeSearchResultReason::PrefixMismatchOrInsufficientBytes => {
341                        match (is_start, node_prefix_comparison_to_search_key_segment) {
342                            (true, Ordering::Equal | Ordering::Greater) => unsafe {
343                                // SAFETY: Covered by function safety doc
344                                minimum_unchecked(node_ptr)
345                            },
346                            (true, Ordering::Less) => unsafe {
347                                // SAFETY: Covered by function safety doc
348                                let max_leaf_ptr = maximum_unchecked(node_ptr);
349                                let max_leaf = max_leaf_ptr.as_ref();
350                                max_leaf.next?
351                            },
352                            (false, Ordering::Equal | Ordering::Less) => unsafe {
353                                // SAFETY: Covered by function safety doc
354                                maximum_unchecked(node_ptr)
355                            },
356                            (false, Ordering::Greater) => unsafe {
357                                // SAFETY: Covered by function safety doc
358                                let min_leaf_ptr = minimum_unchecked(node_ptr);
359                                let min_leaf = min_leaf_ptr.as_ref();
360                                min_leaf.previous?
361                            },
362                        }
363                    },
364                    InnerNodeSearchResultReason::MissingChild => {
365                        fn access_closest_child<
366                            const PREFIX_LEN: usize,
367                            N: InnerNode<PREFIX_LEN>,
368                        >(
369                            inner_ptr: NodePtr<PREFIX_LEN, N>,
370                            child_bounds: (Bound<u8>, Bound<u8>),
371                            is_start: bool,
372                        ) -> KeyByteAndChild<N::Key, N::Value, PREFIX_LEN> {
373                            // SAFETY: Covered by function safety doc
374                            let inner = unsafe { inner_ptr.as_ref() };
375                            let mut child_it = inner.range(child_bounds);
376                            if is_start {
377                                child_it.next()
378                            } else {
379                                child_it.next_back()
380                            }
381                        }
382
383                        let child_bounds = if is_start {
384                            (bound.map(|_| key_bytes[current_depth]), Bound::Unbounded)
385                        } else {
386                            (Bound::Unbounded, bound.map(|_| key_bytes[current_depth]))
387                        };
388                        let possible_next_child = match node_ptr.to_node_ptr() {
389                            ConcreteNodePtr::Node4(inner_ptr) => {
390                                access_closest_child(inner_ptr, child_bounds, is_start)
391                            },
392                            ConcreteNodePtr::Node16(inner_ptr) => {
393                                access_closest_child(inner_ptr, child_bounds, is_start)
394                            },
395                            ConcreteNodePtr::Node48(inner_ptr) => {
396                                access_closest_child(inner_ptr, child_bounds, is_start)
397                            },
398                            ConcreteNodePtr::Node256(inner_ptr) => {
399                                access_closest_child(inner_ptr, child_bounds, is_start)
400                            },
401                            ConcreteNodePtr::LeafNode(_) => unreachable!(
402                                "This branch is unreachable because the \
403                                 TerminatingNodeSearchResult had the value InnerNode"
404                            ),
405                        };
406                        let (_, next_child) = possible_next_child?;
407
408                        if is_start {
409                            // SAFETY: Covered by function safety doc
410                            unsafe { minimum_unchecked(next_child) }
411                        } else {
412                            // SAFETY: Covered by function safety doc
413                            unsafe { maximum_unchecked(next_child) }
414                        }
415                    },
416                },
417            }
418        },
419        Bound::Unbounded => {
420            if is_start {
421                tree.min_leaf
422            } else {
423                tree.max_leaf
424            }
425        },
426    })
427}

429macro_rules! implement_range_iter {
430    (
431        $(#[$outer:meta])*
432        struct $name:ident {
433            tree: $tree_ty:ty,
434            item: $item_ty:ty,
435            $leaf_accessor_func:ident
436        }
437    ) => {
438        $(#[$outer])*
439        pub struct $name<'a, K, V, const PREFIX_LEN: usize = DEFAULT_PREFIX_LEN, A: Allocator = Global> {
440            inner: RawIterator<K, V, PREFIX_LEN>,
441            _tree: $tree_ty,
442        }
443
444        impl<'a, K: AsBytes, V, const PREFIX_LEN: usize, A: Allocator> $name<'a, K, V, PREFIX_LEN, A> {
445            /// Create a new range iterator over the given tree, starting and ending
446            /// according to the given bounds.
447            ///
448            /// # Panics
449            ///
450            /// This function will panic if `start_bound` is greater than `end_bound`.
451            pub(crate) fn new(
452                tree: $tree_ty,
453                start_bound: Bound<&[u8]>,
454                end_bound: Bound<&[u8]>,
455            ) -> Self {
456                let Some(tree_state) = &tree.state else {
457                    return Self {
458                        _tree: tree,
459                        inner: RawIterator::empty(),
460                    };
461                };
462
463                // Validation happens after the empty check so we can avoid panicking for invalid range
464                // when the tree is empty. I believe this matches BTreeMap behavior.
465                validate_range_bounds(&start_bound, &end_bound);
466
467                // SAFETY: Since we have a (shared or mutable) reference to the original TreeMap, we know
468                // there will be no concurrent mutation
469                let possible_start =
470                    unsafe { find_leaf_pointer_for_bound(tree_state, start_bound, true) };
471                let Some(start) = possible_start else {
472                    return Self {
473                        _tree: tree,
474                        inner: RawIterator::empty(),
475                    };
476                };
477
478                // SAFETY: Since we have a (shared or mutable) reference to the original TreeMap, we know
479                // there will be no concurrent mutation
480                let possible_end =
481                    unsafe { find_leaf_pointer_for_bound(tree_state, end_bound, false) };
482                let Some(end) = possible_end else {
483                    return Self {
484                        _tree: tree,
485                        inner: RawIterator::empty(),
486                    };
487                };
488
489                // SAFETY: Since we have a (shared or mutable) reference to the original TreeMap, we know
490                // there will be no concurrent mutation. Also, the reference lifetimes created
491                // are bounded to this `unsafe` block, and don't overlap with any mutation.
492                unsafe {
493                    let start_leaf_bytes = start.as_key_ref().as_bytes();
494                    let end_leaf_bytes = end.as_key_ref().as_bytes();
495
496                    if start_leaf_bytes > end_leaf_bytes {
497                        // Resolved start leaf is not less than or equal to resolved end leaf for
498                        // iteration order
499                        return Self {
500                            _tree: tree,
501                            inner: RawIterator::empty(),
502                        };
503                    }
504                }
505
506                Self {
507                    _tree: tree,
508                    // SAFETY: `start` is guaranteed to be less than or equal to `end` in the iteration
509                    // order because of the check we do on the bytes of the resolved leaf pointers, just
510                    // above this line
511                    inner: unsafe { RawIterator::new(start, end) },
512                }
513            }
514        }
515
516        impl<'a, K, V, const PREFIX_LEN: usize, A: Allocator> Iterator for $name<'a, K, V, PREFIX_LEN, A> {
517            type Item = $item_ty;
518
519            fn next(&mut self) -> Option<Self::Item> {
520                // SAFETY: This iterator has a reference (either shared or mutable) to the
521                // original `TreeMap` it is iterating over, preventing any other modification.
522                let leaf_ptr = unsafe { self.inner.next()? };
523
524                // SAFETY: THe lifetimes returned from this function are returned as bounded by
525                // lifetime 'a, meaning that they cannot outlive this iterator's reference
526                // (shared or mutable) to the original TreeMap.
527                Some(unsafe { leaf_ptr.$leaf_accessor_func() })
528            }
529        }
530
531        impl<'a, K, V, const PREFIX_LEN: usize, A: Allocator> DoubleEndedIterator
532            for $name<'a, K, V, PREFIX_LEN, A>
533        {
534            fn next_back(&mut self) -> Option<Self::Item> {
535                // SAFETY: This iterator has a reference (either shared or mutable) to the
536                // original `TreeMap` it is iterating over, preventing any other modification.
537                let leaf_ptr = unsafe { self.inner.next_back()? };
538
539                // SAFETY: THe lifetimes returned from this function are returned as bounded by
540                // lifetime 'a, meaning that they cannot outlive this iterator's reference
541                // (shared or mutable) to the original TreeMap.
542                Some(unsafe { leaf_ptr.$leaf_accessor_func() })
543            }
544        }
545
546        impl<'a, K, V, const PREFIX_LEN: usize, A: Allocator> FusedIterator for $name<'a, K, V, PREFIX_LEN, A> {}
547    };
548}
