6/// Assuming all the elements are initialized, get a slice to them.
7///
8/// # Safety
9///
10/// It is up to the caller to guarantee that the `MaybeUninit<T>` elements
11/// really are in an initialized state.
12/// Calling this when the content is not yet fully initialized causes
13/// undefined behavior.
14///
15/// See [`assume_init_ref`][core::mem::MaybeUninit::assume_init_ref] for more
16/// details and examples.
17///
18/// **This is a unstable API copied from the Rust standard library, tracking
19/// issue is [#63569][issue-63569]**
20///
21/// [issue-63569]: https://github.com/rust-lang/rust/issues/63569
22#[inline]
23pub const unsafe fn maybe_uninit_slice_assume_init_ref<T>(
24    slice: &[core::mem::MaybeUninit<T>],
25) -> &[T] {
26    #[cfg(feature = "nightly")]
27    {
28        // SAFETY: Covered by condition of containing function
29        unsafe { slice.assume_init_ref() }
30    }
31
32    #[cfg(not(feature = "nightly"))]
33    {
34        // SAFETY: casting `slice` to a `*const [T]` is safe since the caller guarantees
35        // that `slice` is initialized, and `MaybeUninit` is guaranteed to have the same
36        // layout as `T`. The pointer obtained is valid since it refers to memory owned
37        // by `slice` which is a reference and thus guaranteed to be valid for
38        // reads.
39        unsafe { &*(slice as *const [core::mem::MaybeUninit<T>] as *const [T]) }
40    }
41}

43/// Writes a length prefix into this hasher, as part of being prefix-free.
44///
45/// If you're implementing [`Hash`] for a custom collection, call this before
46/// writing its contents to this `Hasher`.  That way
47/// `(collection![1, 2, 3], collection![4, 5])` and
48/// `(collection![1, 2], collection![3, 4, 5])` will provide different
49/// sequences of values to the `Hasher`
50///
51/// The `impl<T> Hash for [T]` includes a call to this method, so if you're
52/// hashing a slice (or array or vector) via its `Hash::hash` method,
53/// you should **not** call this yourself.
54///
55/// This method is only for providing domain separation.  If you want to
56/// hash a `usize` that represents part of the *data*, then it's important
57/// that you pass it to [`Hasher::write_usize`][core::hash::Hasher::write_usize]
58/// instead of to this method.
59///
60/// # Note to Implementers
61///
62/// If you've decided that your `Hasher` is willing to be susceptible to
63/// Hash-DoS attacks, then you might consider skipping hashing some or all
64/// of the `len` provided in the name of increased performance.
65///
66/// **This is a unstable API copied from the Rust standard library, tracking
67/// issue is [#96762][issue-96762]**
68///
69/// [issue-96762]: https://github.com/rust-lang/rust/issues/96762
70#[inline]
71pub fn hasher_write_length_prefix<H: core::hash::Hasher>(state: &mut H, num_entries: usize) {
72    #[cfg(feature = "nightly")]
73    {
74        <H as core::hash::Hasher>::write_length_prefix(state, num_entries);
75    }
76
77    #[cfg(not(feature = "nightly"))]
78    {
79        state.write_usize(num_entries);
80    }
81}

157pub(crate) mod ptr {
158    //! This module contains shim functions for strict-provenance stuff related
159    //! to pointers
160
161    use core::{num::NonZeroUsize, ptr::NonNull};
162
163    #[inline]
164    #[cfg(not(feature = "nightly"))]
165    #[expect(clippy::transmutes_expressible_as_ptr_casts)]
166    pub fn mut_addr<T>(ptr: *mut T) -> usize {
167        // FIXME(strict_provenance_magic): I am magic and should be a compiler
168        // intrinsic. SAFETY: Pointer-to-integer transmutes are valid (if you
169        // are okay with losing the provenance).
170        unsafe { core::mem::transmute(ptr.cast::<()>()) }
171    }
172
173    #[inline]
174    #[cfg(feature = "nightly")]
175    pub fn mut_addr<T>(ptr: *mut T) -> usize {
176        ptr.addr()
177    }
178
179    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
180    #[inline]
181    #[cfg(not(feature = "nightly"))]
182    pub fn mut_with_addr<T>(ptr: *mut T, addr: usize) -> *mut T {
183        let self_addr = mut_addr(ptr) as isize;
184        let dest_addr = addr as isize;
185        let offset = dest_addr.wrapping_sub(self_addr);
186
187        ptr.wrapping_byte_offset(offset)
188    }
189
190    #[cfg(test)]
191    #[inline]
192    #[cfg(not(feature = "nightly"))]
193    pub fn const_addr<T>(ptr: *const T) -> usize {
194        // FIXME(strict_provenance_magic): I am magic and should be a compiler
195        // intrinsic. SAFETY: Pointer-to-integer transmutes are valid (if you
196        // are okay with losing the provenance).
197        ptr.cast::<()>() as usize
198    }
199
200    #[cfg(test)]
201    #[inline]
202    #[cfg(feature = "nightly")]
203    pub fn const_addr<T>(ptr: *const T) -> usize {
204        ptr.addr()
205    }
206
207    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
208    #[inline]
209    #[cfg(not(feature = "nightly"))]
210    pub fn nonnull_addr<T>(ptr: NonNull<T>) -> NonZeroUsize {
211        unsafe { NonZeroUsize::new_unchecked(mut_addr(ptr.as_ptr())) }
212    }
213
214    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
215    #[inline]
216    #[cfg(feature = "nightly")]
217    pub fn nonnull_addr<T>(ptr: NonNull<T>) -> NonZeroUsize {
218        ptr.addr()
219    }
220
221    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
222    #[inline]
223    #[cfg(not(feature = "nightly"))]
224    pub fn nonnull_with_addr<T>(ptr: NonNull<T>, addr: NonZeroUsize) -> NonNull<T> {
225        unsafe { NonNull::new_unchecked(mut_with_addr(ptr.as_ptr(), addr.get())) }
226    }
227
228    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
229    #[inline]
230    #[cfg(feature = "nightly")]
231    pub fn nonnull_with_addr<T>(ptr: NonNull<T>, addr: NonZeroUsize) -> NonNull<T> {
232        ptr.with_addr(addr)
233    }
234
235    #[inline]
236    #[cfg(not(feature = "nightly"))]
237    pub fn nonnull_map_addr<T>(
238        ptr: NonNull<T>,
239        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
240    ) -> NonNull<T> {
241        let ptr = ptr.as_ptr();
242        let old_addr = mut_addr(ptr);
243        let new_addr = f(unsafe {
244            // SAFETY: `ptr` was a `NonNull` pointer, the address was required to be
245            // non-zero.
246            NonZeroUsize::new_unchecked(old_addr)
247        });
248
249        let offset = new_addr.get().wrapping_sub(old_addr);
250        // This is the canonical desugaring of this operation
251        let new_ptr = ptr.wrapping_byte_offset(offset as isize);
252
253        // SAFETY: The 2 lines of code above are just changing the address of the
254        // pointer, while trying to retain the same provenance. Since the `new_addr`
255        // returned from `f` is non-zero, then the `new_ptr` must be non-null.
256        unsafe { NonNull::new_unchecked(new_ptr) }
257    }
258
259    #[inline]
260    #[cfg(feature = "nightly")]
261    pub fn nonnull_map_addr<T>(
262        ptr: NonNull<T>,
263        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
264    ) -> NonNull<T> {
265        ptr.map_addr(f)
266    }
