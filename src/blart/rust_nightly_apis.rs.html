4/// Assuming all the elements are initialized, get a slice to them.
5///
6/// # Safety
7///
8/// It is up to the caller to guarantee that the `MaybeUninit<T>` elements
9/// really are in an initialized state.
10/// Calling this when the content is not yet fully initialized causes
11/// undefined behavior.
12///
13/// See [`assume_init_ref`][core::mem::MaybeUninit::assume_init_ref] for more
14/// details and examples.
15///
16/// **This is a unstable API copied from the Rust standard library, tracking
17/// issue is [#63569][issue-63569]**
18///
19/// [issue-63569]: https://github.com/rust-lang/rust/issues/63569
20#[inline]
21pub const unsafe fn maybe_uninit_slice_assume_init_ref<T>(
22    slice: &[core::mem::MaybeUninit<T>],
23) -> &[T] {
24    #[cfg(feature = "nightly")]
25    {
26        // SAFETY: Covered by condition of containing function
27        unsafe { slice.assume_init_ref() }
28    }
29
30    #[cfg(not(feature = "nightly"))]
31    {
32        // SAFETY: casting `slice` to a `*const [T]` is safe since the caller guarantees
33        // that `slice` is initialized, and `MaybeUninit` is guaranteed to have the same
34        // layout as `T`. The pointer obtained is valid since it refers to memory owned
35        // by `slice` which is a reference and thus guaranteed to be valid for
36        // reads.
37        unsafe { &*(slice as *const [core::mem::MaybeUninit<T>] as *const [T]) }
38    }
39}

41/// Assuming all the elements are initialized, get a mutable slice to them.
42///
43/// # Safety
44///
45/// It is up to the caller to guarantee that the `MaybeUninit<T>` elements
46/// really are in an initialized state.
47/// Calling this when the content is not yet fully initialized causes
48/// undefined behavior.
49///
50/// See [`assume_init_mut`][core::mem::MaybeUninit::assume_init_mut] for more
51/// details and examples.
52///
53/// **This is a unstable API copied from the Rust standard library, tracking
54/// issue is [#63569][issue-63569]**
55///
56/// [issue-63569]: https://github.com/rust-lang/rust/issues/63569
57#[inline]
58pub unsafe fn maybe_uninit_slice_assume_init_mut<T>(
59    slice: &mut [core::mem::MaybeUninit<T>],
60) -> &mut [T] {
61    #[cfg(feature = "nightly")]
62    {
63        // SAFETY: Covered by condition of containing function
64        unsafe { slice.assume_init_mut() }
65    }
66
67    #[cfg(not(feature = "nightly"))]
68    {
69        // SAFETY: similar to safety notes for `maybe_uninit_slice_assume_init_ref`, but
70        // we have a mutable reference which is also guaranteed to be valid for
71        // writes.
72        unsafe { &mut *(slice as *mut [core::mem::MaybeUninit<T>] as *mut [T]) }
73    }
74}

76/// Writes a length prefix into this hasher, as part of being prefix-free.
77///
78/// If you're implementing [`Hash`] for a custom collection, call this before
79/// writing its contents to this `Hasher`.  That way
80/// `(collection![1, 2, 3], collection![4, 5])` and
81/// `(collection![1, 2], collection![3, 4, 5])` will provide different
82/// sequences of values to the `Hasher`
83///
84/// The `impl<T> Hash for [T]` includes a call to this method, so if you're
85/// hashing a slice (or array or vector) via its `Hash::hash` method,
86/// you should **not** call this yourself.
87///
88/// This method is only for providing domain separation.  If you want to
89/// hash a `usize` that represents part of the *data*, then it's important
90/// that you pass it to [`Hasher::write_usize`][core::hash::Hasher::write_usize]
91/// instead of to this method.
92///
93/// # Note to Implementers
94///
95/// If you've decided that your `Hasher` is willing to be susceptible to
96/// Hash-DoS attacks, then you might consider skipping hashing some or all
97/// of the `len` provided in the name of increased performance.
98///
99/// **This is a unstable API copied from the Rust standard library, tracking
100/// issue is [#96762][issue-96762]**
101///
102/// [issue-96762]: https://github.com/rust-lang/rust/issues/96762
103#[inline]
104pub fn hasher_write_length_prefix<H: core::hash::Hasher>(state: &mut H, num_entries: usize) {
105    #[cfg(feature = "nightly")]
106    {
107        <H as core::hash::Hasher>::write_length_prefix(state, num_entries);
108    }
109
110    #[cfg(not(feature = "nightly"))]
111    {
112        state.write_usize(num_entries);
113    }
114}

190pub(crate) mod ptr {
191    //! This module contains shim functions for strict-provenance stuff related
192    //! to pointers
193
194    use core::{num::NonZeroUsize, ptr::NonNull};
195
196    #[inline]
197    #[cfg(not(feature = "nightly"))]
198    #[expect(clippy::transmutes_expressible_as_ptr_casts)]
199    pub fn mut_addr<T>(ptr: *mut T) -> usize {
200        // FIXME(strict_provenance_magic): I am magic and should be a compiler
201        // intrinsic. SAFETY: Pointer-to-integer transmutes are valid (if you
202        // are okay with losing the provenance).
203        unsafe { core::mem::transmute(ptr.cast::<()>()) }
204    }
205
206    #[inline]
207    #[cfg(feature = "nightly")]
208    pub fn mut_addr<T>(ptr: *mut T) -> usize {
209        ptr.addr()
210    }
211
212    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
213    #[inline]
214    #[cfg(not(feature = "nightly"))]
215    pub fn mut_with_addr<T>(ptr: *mut T, addr: usize) -> *mut T {
216        let self_addr = mut_addr(ptr) as isize;
217        let dest_addr = addr as isize;
218        let offset = dest_addr.wrapping_sub(self_addr);
219
220        ptr.wrapping_byte_offset(offset)
221    }
222
223    #[cfg(test)]
224    #[inline]
225    #[cfg(not(feature = "nightly"))]
226    pub fn const_addr<T>(ptr: *const T) -> usize {
227        // FIXME(strict_provenance_magic): I am magic and should be a compiler
228        // intrinsic. SAFETY: Pointer-to-integer transmutes are valid (if you
229        // are okay with losing the provenance).
230        ptr.cast::<()>() as usize
231    }
232
233    #[cfg(test)]
234    #[inline]
235    #[cfg(feature = "nightly")]
236    pub fn const_addr<T>(ptr: *const T) -> usize {
237        ptr.addr()
238    }

240    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
241    #[inline]
242    #[cfg(not(feature = "nightly"))]
243    pub fn nonnull_addr<T>(ptr: NonNull<T>) -> NonZeroUsize {
244        unsafe { NonZeroUsize::new_unchecked(mut_addr(ptr.as_ptr())) }
245    }
246
247    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
248    #[inline]
249    #[cfg(feature = "nightly")]
250    pub fn nonnull_addr<T>(ptr: NonNull<T>) -> NonZeroUsize {
251        ptr.addr()
252    }
253
254    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
255    #[inline]
256    #[cfg(not(feature = "nightly"))]
257    pub fn nonnull_with_addr<T>(ptr: NonNull<T>, addr: NonZeroUsize) -> NonNull<T> {
258        unsafe { NonNull::new_unchecked(mut_with_addr(ptr.as_ptr(), addr.get())) }
259    }
260
261    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
262    #[inline]
263    #[cfg(feature = "nightly")]
264    pub fn nonnull_with_addr<T>(ptr: NonNull<T>, addr: NonZeroUsize) -> NonNull<T> {
265        ptr.with_addr(addr)
266    }
267
268    #[inline]
269    #[cfg(not(feature = "nightly"))]
270    pub fn nonnull_map_addr<T>(
271        ptr: NonNull<T>,
272        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
273    ) -> NonNull<T> {
274        let ptr = ptr.as_ptr();
275        let old_addr = mut_addr(ptr);
276        let new_addr = f(unsafe {
277            // SAFETY: `ptr` was a `NonNull` pointer, the address was required to be
278            // non-zero.
279            NonZeroUsize::new_unchecked(old_addr)
280        });
281
282        let offset = new_addr.get().wrapping_sub(old_addr);
283        // This is the canonical desugaring of this operation
284        let new_ptr = ptr.wrapping_byte_offset(offset as isize);
285
286        // SAFETY: The 2 lines of code above are just changing the address of the
287        // pointer, while trying to retain the same provenance. Since the `new_addr`
288        // returned from `f` is non-zero, then the `new_ptr` must be non-null.
289        unsafe { NonNull::new_unchecked(new_ptr) }
290    }
291
292    #[inline]
293    #[cfg(feature = "nightly")]
294    pub fn nonnull_map_addr<T>(
295        ptr: NonNull<T>,
296        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
297    ) -> NonNull<T> {
298        ptr.map_addr(f)
299    }
