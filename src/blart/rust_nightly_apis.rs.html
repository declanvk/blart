4/// Assuming all the elements are initialized, get a slice to them.
5///
6/// # Safety
7///
8/// It is up to the caller to guarantee that the `MaybeUninit<T>` elements
9/// really are in an initialized state.
10/// Calling this when the content is not yet fully initialized causes
11/// undefined behavior.
12///
13/// See [`assume_init_ref`][std::mem::MaybeUninit::assume_init_ref] for more
14/// details and examples.
15///
16/// **This is a unstable API copied from the Rust standard library, tracking
17/// issue is [#63569][issue-63569]**
18///
19/// [issue-63569]: https://github.com/rust-lang/rust/issues/63569
20#[inline]
21pub const unsafe fn maybe_uninit_slice_assume_init_ref<T>(
22    slice: &[std::mem::MaybeUninit<T>],
23) -> &[T] {
24    #[cfg(feature = "nightly")]
25    {
26        // SAFETY: Covered by condition of containing function
27        unsafe { slice.assume_init_ref() }
28    }
29
30    #[cfg(not(feature = "nightly"))]
31    {
32        // SAFETY: casting `slice` to a `*const [T]` is safe since the caller guarantees
33        // that `slice` is initialized, and `MaybeUninit` is guaranteed to have the same
34        // layout as `T`. The pointer obtained is valid since it refers to memory owned
35        // by `slice` which is a reference and thus guaranteed to be valid for
36        // reads.
37        unsafe { &*(slice as *const [std::mem::MaybeUninit<T>] as *const [T]) }
38    }
39}

41/// Assuming all the elements are initialized, get a mutable slice to them.
42///
43/// # Safety
44///
45/// It is up to the caller to guarantee that the `MaybeUninit<T>` elements
46/// really are in an initialized state.
47/// Calling this when the content is not yet fully initialized causes
48/// undefined behavior.
49///
50/// See [`assume_init_mut`][std::mem::MaybeUninit::assume_init_mut] for more
51/// details and examples.
52///
53/// **This is a unstable API copied from the Rust standard library, tracking
54/// issue is [#63569][issue-63569]**
55///
56/// [issue-63569]: https://github.com/rust-lang/rust/issues/63569
57#[inline]
58pub unsafe fn maybe_uninit_slice_assume_init_mut<T>(
59    slice: &mut [std::mem::MaybeUninit<T>],
60) -> &mut [T] {
61    #[cfg(feature = "nightly")]
62    {
63        // SAFETY: Covered by condition of containing function
64        unsafe { slice.assume_init_mut() }
65    }
66
67    #[cfg(not(feature = "nightly"))]
68    {
69        // SAFETY: similar to safety notes for `maybe_uninit_slice_assume_init_ref`, but
70        // we have a mutable reference which is also guaranteed to be valid for
71        // writes.
72        unsafe { &mut *(slice as *mut [std::mem::MaybeUninit<T>] as *mut [T]) }
73    }
74}

76/// Writes a length prefix into this hasher, as part of being prefix-free.
77///
78/// If you're implementing [`Hash`] for a custom collection, call this before
79/// writing its contents to this `Hasher`.  That way
80/// `(collection![1, 2, 3], collection![4, 5])` and
81/// `(collection![1, 2], collection![3, 4, 5])` will provide different
82/// sequences of values to the `Hasher`
83///
84/// The `impl<T> Hash for [T]` includes a call to this method, so if you're
85/// hashing a slice (or array or vector) via its `Hash::hash` method,
86/// you should **not** call this yourself.
87///
88/// This method is only for providing domain separation.  If you want to
89/// hash a `usize` that represents part of the *data*, then it's important
90/// that you pass it to [`Hasher::write_usize`][std::hash::Hasher::write_usize]
91/// instead of to this method.
92///
93/// # Note to Implementers
94///
95/// If you've decided that your `Hasher` is willing to be susceptible to
96/// Hash-DoS attacks, then you might consider skipping hashing some or all
97/// of the `len` provided in the name of increased performance.
98///
99/// **This is a unstable API copied from the Rust standard library, tracking
100/// issue is [#96762][issue-96762]**
101///
102/// [issue-96762]: https://github.com/rust-lang/rust/issues/96762
103#[inline]
104pub fn hasher_write_length_prefix<H: std::hash::Hasher>(state: &mut H, num_entries: usize) {
105    #[cfg(feature = "nightly")]
106    {
107        <H as std::hash::Hasher>::write_length_prefix(state, num_entries);
108    }
109
110    #[cfg(not(feature = "nightly"))]
111    {
112        state.write_usize(num_entries);
113    }
114}

116/// Constructs a new boxed slice with uninitialized contents.
117///
118/// **This is a unstable API copied from the Rust standard library, tracking
119/// issue is [#63291][issue-63291]**
120///
121/// [issue-63291]: https://github.com/rust-lang/rust/issues/63291
122#[inline]
123pub fn box_new_uninit_slice<T>(len: usize) -> Box<[std::mem::MaybeUninit<T>]> {
124    #[cfg(feature = "nightly")]
125    {
126        Box::new_uninit_slice(len)
127    }
128
129    #[cfg(not(feature = "nightly"))]
130    {
131        Vec::from_iter((0..len).map(|_| std::mem::MaybeUninit::uninit())).into_boxed_slice()
132    }
133}

135/// Informs the optimizer that a condition is always true.
136/// If the condition is false, the behavior is undefined.
137///
138/// No code is generated for this intrinsic, but the optimizer will try
139/// to preserve it (and its condition) between passes, which may interfere
140/// with optimization of surrounding code and reduce performance. It should
141/// not be used if the invariant can be discovered by the optimizer on its
142/// own, or if it does not enable any significant optimizations.
143///
144/// This intrinsic does not have a stable counterpart.
145///
146/// **This is a unstable API copied from the Rust standard library**
147macro_rules! assume {
148    ($b:expr) => {
149        debug_assert!($b);
150        #[cfg(feature = "nightly")]
151        std::intrinsics::assume($b)
152    };
153}

155pub(crate) use assume;

242    //! This module contains shim functions for strict-provenance stuff related
243    //! to pointers

245    use std::{num::NonZeroUsize, ptr::NonNull};

247    #[inline]
248    #[cfg(not(feature = "nightly"))]
249    #[allow(clippy::transmutes_expressible_as_ptr_casts)]
250    pub fn mut_addr<T>(ptr: *mut T) -> usize {
251        // FIXME(strict_provenance_magic): I am magic and should be a compiler
252        // intrinsic. SAFETY: Pointer-to-integer transmutes are valid (if you
253        // are okay with losing the provenance).
254        unsafe { std::mem::transmute(ptr.cast::<()>()) }
255    }
256
257    #[inline]
258    #[cfg(feature = "nightly")]
259    pub fn mut_addr<T>(ptr: *mut T) -> usize {
260        ptr.addr()
261    }
262
263    #[cfg(all(test, any(feature = "allocator-api2", feature = "nightly")))]
264    #[inline]
265    #[cfg(not(feature = "nightly"))]
266    pub fn mut_with_addr<T>(ptr: *mut T, addr: usize) -> *mut T {
267        let self_addr = mut_addr(ptr) as isize;
268        let dest_addr = addr as isize;
269        let offset = dest_addr.wrapping_sub(self_addr);
270
271        ptr.wrapping_byte_offset(offset)
272    }

326    #[inline]
327    #[cfg(not(feature = "nightly"))]
328    pub fn nonnull_map_addr<T>(
329        ptr: NonNull<T>,
330        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
331    ) -> NonNull<T> {
332        let ptr = ptr.as_ptr();
333        let old_addr = mut_addr(ptr);
334        let new_addr = f(unsafe {
335            // SAFETY: `ptr` was a `NonNull` pointer, the address was required to be
336            // non-zero.
337            NonZeroUsize::new_unchecked(old_addr)
338        });
339
340        let offset = new_addr.get().wrapping_sub(old_addr);
341        // This is the canonical desugaring of this operation
342        let new_ptr = ptr.wrapping_byte_offset(offset as isize);
343
344        // SAFETY: The 2 lines of code above are just changing the address of the
345        // pointer, while trying to retain the same provenance. Since the `new_addr`
346        // returned from `f` is non-zero, then the `new_ptr` must be non-null.
347        unsafe { NonNull::new_unchecked(new_ptr) }
348    }
349
350    #[inline]
351    #[cfg(feature = "nightly")]
352    pub fn nonnull_map_addr<T>(
353        ptr: NonNull<T>,
354        f: impl FnOnce(NonZeroUsize) -> NonZeroUsize,
355    ) -> NonNull<T> {
356        ptr.map_addr(f)
357    }
