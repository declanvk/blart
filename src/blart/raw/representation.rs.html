29/// The representation of inner nodes
30#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
31#[repr(u8)]
32pub enum NodeType {
33    /// Node that references between 2 and 4 children
34    Node4 = 0b000,
35    /// Node that references between 5 and 16 children
36    Node16 = 0b001, // 0b001
37    /// Node that references between 17 and 48 children
38    Node48 = 0b010, // 0b010
39    /// Node that references between 49 and 256 children
40    Node256 = 0b011, // 0b011
41    /// Node that contains a single value
42    Leaf = 0b100, // 0b100
43}
44
45impl NodeType {
46    /// Converts a u8 value to a [`NodeType`]
47    ///
48    /// # Safety
49    ///  - `src` must be a valid variant from the enum
50    const unsafe fn from_u8(src: u8) -> NodeType {
51        // SAFETY: `NodeType` is repr(u8)
52        unsafe { core::mem::transmute::<u8, NodeType>(src) }
53    }
54
55    /// Return true if an [`InnerNode`] with the given [`NodeType`] and
56    /// specified number of children should be shrunk.
57    ///
58    /// # Panics
59    ///  - Panics if `node_type` equals [`NodeType::Leaf`]
60    pub fn should_shrink_inner_node(self, num_children: usize) -> bool {
61        match self {
62            NodeType::Node4 => false,
63            NodeType::Leaf => panic!("cannot shrink leaf"),
64            _ => num_children < *self.capacity_range().start(),
65        }
66    }
67
68    /// Return the range of number of children that each node type accepts.
69    pub const fn capacity_range(self) -> RangeInclusive<usize> {
70        match self {
71            NodeType::Node4 => 2..=4,
72            NodeType::Node16 => 5..=16,
73            NodeType::Node48 => 17..=48,
74            NodeType::Node256 => 49..=256,
75            NodeType::Leaf => 0..=0,
76        }
77    }
78}

191/// This trait contains all the functions/types that every inner node must
192/// implement.
193///
194/// This doesn't necessarily mean that the specific type will be used as an
195/// inner node, since that requires assigning it a [`NodeType`] and determining
196/// its "grow"/"shrink" order. See [`InnerNode`] for those details.
197///
198/// # Safety
199///
200/// All structures that implement this trait must be `repr(C)` and have a
201/// [`Header`] as the first field of the struct.
202pub unsafe trait InnerNodeCommon<K, V, const PREFIX_LEN: usize>: Sized {
203    /// The type of the iterator over all children of the inner node
204    type Iter<'a>: Iterator<Item = (u8, OpaqueNodePtr<K, V, PREFIX_LEN>)>
205        + DoubleEndedIterator
206        + FusedIterator
207    where
208        Self: 'a;
209
210    /// Create an empty [`InnerNode`], with no children and no prefix
211    #[inline]
212    fn empty() -> Self {
213        Self::from_header(Header::empty())
214    }
215
216    /// Create a new [`InnerNode`] using `prefix` as the node prefix and
217    /// `prefix_len` as the node prefix length.
218    ///
219    /// This is done because when a prefix mismatch happens
220    /// the length of the mismatch can be grater or equal to
221    /// prefix size, since we search for the first child of the
222    /// node to recreate the prefix, that's why we don't use
223    /// `prefix.len()` as the node prefix length
224    fn from_prefix(prefix: &[u8], prefix_len: usize) -> Self {
225        Self::from_header(Header::new(prefix, prefix_len))
226    }
227
228    /// Create a new [`InnerNode`] using a `Header`
229    fn from_header(header: Header<PREFIX_LEN>) -> Self;
230
231    /// Get the `Header` from the [`InnerNode`]
232    fn header(&self) -> &Header<PREFIX_LEN>;
233
234    /// Search through this node for a child node that corresponds to the given
235    /// key fragment.
236    fn lookup_child(&self, key_fragment: u8) -> Option<OpaqueNodePtr<K, V, PREFIX_LEN>>;
237
238    /// Write a child pointer with key fragment to this inner node.
239    ///
240    /// If the key fragment already exists in the node, overwrite the existing
241    /// child pointer.
242    ///
243    /// # Panics
244    ///  - Panics when the node is full.
245    fn write_child(&mut self, key_fragment: u8, child_pointer: OpaqueNodePtr<K, V, PREFIX_LEN>);
246
247    /// Attempt to remove a child pointer at the key fragment from this inner
248    /// node.
249    ///
250    /// If the key fragment does not exist in this node, return `None`.
251    fn remove_child(&mut self, key_fragment: u8) -> Option<OpaqueNodePtr<K, V, PREFIX_LEN>>;
252
253    /// Create an iterator over all `(key bytes, child pointers)` in this inner
254    /// node.
255    fn iter(&self) -> Self::Iter<'_>;
256
257    /// Create an iterator over a subset of `(key bytes, child pointers)`, using
258    /// the given `bound` as a restriction on the set of key bytes.
259    fn range(
260        &self,
261        bound: impl RangeBounds<u8>,
262    ) -> impl DoubleEndedIterator<Item = (u8, OpaqueNodePtr<K, V, PREFIX_LEN>)> + FusedIterator;
263
264    /// Test the given key against the inner node header prefix by checking that
265    /// the key length is greater than or equal to the length of the header
266    /// prefix.
267    ///
268    /// The `truncated_key` argument should be the overall key bytes shortened
269    /// to the current depth.
270    ///
271    /// This is called "optimistic" matching, because it assumes that there will
272    /// not be a mismatch in the contents of the header prefix when compared to
273    /// the key. The caller who uses this function must perform a final check
274    /// against the leaf key bytes to make sure that the search key matches the
275    /// found key.
276    ///
277    /// # Errors
278    /// This function returns an error if there is a mismatch between the
279    /// partial stored prefix and the given truncated key.
280    #[inline]
281    fn optimistic_match_prefix(
282        &self,
283        truncated_key: &[u8],
284    ) -> Result<PrefixMatch, OptimisticMismatch> {
285        if truncated_key.len() < self.header().prefix_len() {
286            Err(OptimisticMismatch {
287                matched_bytes: truncated_key.len(),
288            })
289        } else {
290            Ok(PrefixMatch {
291                matched_bytes: self.header().prefix_len(),
292            })
293        }
294    }
295
296    /// Test the given key against the inner node header prefix by comparing the
297    /// bytes.
298    ///
299    /// The `truncated_key` argument should be the overall key bytes shortened
300    /// to the current depth.
301    ///
302    /// If the length of the header prefix is greater than the number of bytes
303    /// stored (there are implicit bytes), then this falls back to using
304    /// [`optimistic_match_prefix`][InnerNodeCommon::optimistic_match_prefix].
305    ///
306    /// If this function fell into that condition, then the `any_implicit_bytes`
307    /// flag will be set to `true` in the `Ok` case and `prefix_byte` will be
308    /// set to `None` in the `Err` case.
309    ///
310    /// If either of those conditions are true, and the caller of this function
311    /// reaches a leaf node using these results, then the caller must perform a
312    /// final check against the leaf key bytes to make sure that the search
313    /// key matches the found key.
314    ///
315    /// # Errors
316    /// This function returns an error if there is a mismatch between the given
317    /// truncated key and the stored (implicit or full) prefix.
318    #[inline]
319    fn attempt_pessimistic_match_prefix(
320        &self,
321        truncated_key: &[u8],
322    ) -> Result<AttemptOptimisticPrefixMatch, PessimisticMismatch> {
323        if PREFIX_LEN < self.header().prefix_len() {
324            let PrefixMatch { matched_bytes } = self.optimistic_match_prefix(truncated_key)?;
325
326            Ok(AttemptOptimisticPrefixMatch {
327                matched_bytes,
328                any_implicit_bytes: true,
329            })
330        } else {
331            // All bytes are explicit, this can proceed as normal
332
333            let prefix = self.header().read_prefix();
334
335            let matched_bytes = prefix
336                .iter()
337                .zip(truncated_key)
338                .take_while(|(a, b)| **a == **b)
339                .count();
340            if matched_bytes < self.header().prefix_len() {
341                Err(PessimisticMismatch {
342                    matched_bytes,
343                    prefix_byte: Some(prefix[matched_bytes]),
344                })
345            } else {
346                Ok(AttemptOptimisticPrefixMatch {
347                    matched_bytes,
348                    any_implicit_bytes: false,
349                })
350            }
351        }
352    }
353
354    /// Compares the compressed path of a node with the key and returns the
355    /// number of equal bytes.
356    ///
357    /// This function will read the full prefix for this inner node, even if it
358    /// needs to search a descendant leaf node to find implicit bytes.
359    ///
360    /// # Safety
361    /// `current_depth` must be less than or equal to `key.len()`
362    ///
363    /// # Errors
364    /// This function returns an error if there is a mismatch between the
365    /// provided key and the full prefix.
366    #[inline]
367    unsafe fn match_full_prefix(
368        &self,
369        key: &[u8],
370        current_depth: usize,
371    ) -> Result<PrefixMatch, ExplicitMismatch<K, V, PREFIX_LEN>>
372    where
373        K: AsBytes,
374    {
375        unsafe {
376            // SAFETY: Since we are iterating the key and prefixes, we
377            // expect that the depth never exceeds the key len.
378            // Because if this happens we ran out of bytes in the key to match
379            // and the whole process should be already finished
380            core::hint::assert_unchecked(current_depth <= key.len());
381        }
382
383        let (prefix, leaf_ptr) = self.read_full_prefix(current_depth);
384        let key = &key[current_depth..];
385
386        let matched_bytes = prefix
387            .iter()
388            .zip(key)
389            .take_while(|(a, b)| **a == **b)
390            .count();
391        if matched_bytes < prefix.len() {
392            Err(ExplicitMismatch {
393                matched_bytes,
394                prefix_byte: prefix[matched_bytes],
395                leaf_ptr,
396            })
397        } else {
398            Ok(PrefixMatch { matched_bytes })
399        }
400    }
401
402    /// Read the prefix as a whole, by reconstructing it if necessary from a
403    /// leaf
404    #[inline]
405    fn read_full_prefix<'b>(&'b self, current_depth: usize) -> NodePrefix<'b, K, V, PREFIX_LEN>
406    where
407        K: AsBytes + 'b,
408        V: 'b,
409    {
410        let header = &self.header();
411        let len = header.prefix_len();
412        if likely!(len <= PREFIX_LEN) {
413            (header.read_prefix(), None)
414        } else {
415            // SAFETY: By construction a InnerNode, must have >= 1 children, this
416            // is even more strict since in the case of 1 child the node can be
417            // collapsed, so a InnerNode must have >= 2 children, so it's safe
418            // to search for the minium. And the same applies to the `minimum_unchecked`
419            // function
420            let (_, min_child) = self.min();
421            let leaf_ptr = unsafe { minimum_unchecked(min_child) };
422
423            // SAFETY: Since have a shared reference
424            // is safe to create a shared reference from it
425            let leaf = unsafe { leaf_ptr.as_ref() };
426            let leaf = leaf.key_ref().as_bytes();
427
428            unsafe {
429                // SAFETY: Since we are iterating the key and prefixes, we
430                // expect that the depth never exceeds the key len.
431                // Because if this happens we ran out of bytes in the key to match
432                // and the whole process should be already finished
433                core::hint::assert_unchecked(current_depth <= leaf.len());
434
435                // SAFETY: By the construction of the prefix we know that this is inbounds
436                // since the prefix len guarantees it to us
437                core::hint::assert_unchecked(current_depth + len <= leaf.len());
438
439                // SAFETY: This can't overflow since len comes from a u32
440                core::hint::assert_unchecked(current_depth <= current_depth + len);
441            }
442            let leaf = &leaf[current_depth..(current_depth + len)];
443            (leaf, Some(leaf_ptr))
444        }
445    }
446
447    /// Returns the minimum child pointer from this node and it's key
448    ///
449    /// Since this is a [`InnerNode`] we assume that the we have at least one
450    /// child, (more strictly we have 2, because with one child the node would
451    /// have collapsed) so in this way we can avoid the [`Option`]. This is safe
452    /// because if we had no children this current node should have been
453    /// deleted.
454    fn min(&self) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>);
455
456    /// Returns the maximum child pointer from this node and it's key
457    ///
458    /// Since this is a [`InnerNode`] we assume that the we have at least one
459    /// child, (more strictly we have 2, because with one child the node would
460    /// have collapsed) so in this way we can avoid the [`Option`]. This is safe
461    /// because if we had, no children this current node should have been
462    /// deleted.
463    fn max(&self) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>);
464}

466/// Common methods implemented by all inner nodes that will be used in the tree.
467///
468/// To contrast with [`InnerNodeCommon`], this trait:
469///   - Incorporates a set value of [`NodeType`], accessible via [`Node::TYPE`].
470///   - Specifies where the inner node sits in the "growth order", via
471///     [`InnerNode::GrownNode`] and [`InnerNode::ShrunkNode`].
472pub trait InnerNode<const PREFIX_LEN: usize>:
473    Node<PREFIX_LEN> + Sized + fmt::Debug + InnerNodeCommon<Self::Key, Self::Value, PREFIX_LEN>
474{
475    /// The type of the next larger node type.
476    type GrownNode: InnerNode<PREFIX_LEN, Key = Self::Key, Value = Self::Value>;
477
478    /// The type of the next smaller node type.
479    type ShrunkNode: InnerNode<PREFIX_LEN, Key = Self::Key, Value = Self::Value>;
480
481    /// Grow this node into the next larger class, copying over children and
482    /// prefix information.
483    fn grow(&self) -> Self::GrownNode;
484
485    /// Shrink this node into the next smaller class, copying over children and
486    /// prefix information.
487    ///
488    /// # Panics
489    ///  - Panics if the new, smaller node size does not have enough capacity to
490    ///    hold all the children.
491    fn shrink(&self) -> Self::ShrunkNode;
492
493    /// Returns true if this node has no more space to store children.
494    fn is_full(&self) -> bool {
495        self.header().num_children() >= *Self::TYPE.capacity_range().end()
496    }
497}

509/// This enum represents different kinds of tree paths pointing to a leaf node.
510///
511/// Each variant contains information that is useful when inserting or deleting
512/// a leaf at the implied location.
513pub enum TreePath<K, V, const PREFIX_LEN: usize> {
514    /// This variant indicates a tree with only a single leaf node as root.
515    ///
516    /// It has no grandparent or parent node.
517    Root,
518    /// This variant indicates a tree with a non-leaf root, and the leaf node
519    /// we're pointed to is a direct child of the root.
520    ///
521    /// The leaf has no grandparent node.
522    ChildOfRoot {
523        /// A pointer to the root node of the tree.
524        parent: OpaqueNodePtr<K, V, PREFIX_LEN>,
525        /// The key byte which selects the leaf node when used as lookup in
526        /// the root parent.
527        child_key_byte: u8,
528    },
529    /// This variant covers all other tree cases, where the leaf node has both a
530    /// parent and a grandparent node.
531    Normal {
532        /// A pointer to the grandparent node.
533        grandparent: OpaqueNodePtr<K, V, PREFIX_LEN>,
534        /// The key byte which selects the parent node when used as lookup in
535        /// the grandparent.
536        parent_key_byte: u8,
537        /// A pointer to the parent node.
538        parent: OpaqueNodePtr<K, V, PREFIX_LEN>,
539        /// The key byte which selects the leaf node when used as lookup in
540        /// the parent.
541        child_key_byte: u8,
542    },
543}

557            Self::ChildOfRoot {

560            } => f

564                .finish(),
565            Self::Normal {
566                grandparent,

568                parent,
569                child_key_byte,
570            } => f
571                .debug_struct("Normal")
572                .field("grandparent", grandparent)
573                .field("parent_key_byte", parent_key_byte)
574                .field("parent", parent)
575                .field("child_key_byte", child_key_byte)
576                .finish(),
577        }
578    }

581/// This type is used to track the parent and grandparent when searching down
582/// the tree.
583#[derive(Debug)]
584pub struct TreePathSearch<K, V, const PREFIX_LEN: usize> {
585    current_grandparent: Option<(OpaqueNodePtr<K, V, PREFIX_LEN>, u8)>,
586    current_parent: Option<(OpaqueNodePtr<K, V, PREFIX_LEN>, u8)>,
587}
588
589impl<K, V, const PREFIX_LEN: usize> Default for TreePathSearch<K, V, PREFIX_LEN> {
590    fn default() -> Self {
591        Self {
592            current_grandparent: None,
593            current_parent: None,
594        }
595    }
596}
597
598impl<K, V, const PREFIX_LEN: usize> TreePathSearch<K, V, PREFIX_LEN> {
599    /// Register that the search procedure passed an inner node, and update the
600    /// current parent and grandparent.
601    pub fn visit_inner_node(&mut self, inner_node: OpaqueNodePtr<K, V, PREFIX_LEN>, key_byte: u8) {
602        self.current_grandparent = self.current_parent;
603        self.current_parent = Some((inner_node, key_byte));
604    }
605
606    /// Complete the tree search and return a [`TreePath`] which has the parent
607    /// + grandparent information.
608    pub fn finish(self) -> TreePath<K, V, PREFIX_LEN> {
609        match (self.current_grandparent, self.current_parent) {
610            (None, None) => TreePath::Root,
611            (None, Some((parent, child_key_byte))) => TreePath::ChildOfRoot {
612                parent,
613                child_key_byte,
614            },
615            (Some((grandparent, parent_key_byte)), Some((parent, child_key_byte))) => {
616                TreePath::Normal {
617                    grandparent,
618                    parent_key_byte,
619                    parent,
620                    child_key_byte,
621                }
622            },
623            (Some(_), None) => {
624                unreachable!("Impossible for grandparent to present while parent is not")
625            },
626        }
627    }
628}

630/// Check that the given range bounds are valid, if not then panic.
631///
632/// Validity means:
633///  1. Range bounds are not completely empty
634///  2. Start bound is not greater than end bound
635fn assert_valid_range_bounds(bound: &impl RangeBounds<u8>) {
636    {
637        match (bound.start_bound(), bound.end_bound()) {
638            (Bound::Excluded(s), Bound::Excluded(e)) if s == e => {
639                panic!("range start and end are equal and excluded: ({s:?})")
640            },
641            (Bound::Included(s) | Bound::Excluded(s), Bound::Included(e) | Bound::Excluded(e))
642                if s > e =>
643            {
644                panic!("range start ({s:?}) is greater than range end ({e:?})")
645            },
646            _ => {},
647        }
648    }
649}

659    #[test]
660    #[cfg(target_pointer_width = "64")]
661    fn node_sizes() {
662        const DEFAULT_PREFIX_LEN: usize = 16;
663        assert_eq!(mem::size_of::<Header<DEFAULT_PREFIX_LEN>>(), 24);
664
665        const EXPECTED_HEADER_SIZE: usize = DEFAULT_PREFIX_LEN.next_multiple_of(4) + 8;
666        assert_eq!(EXPECTED_HEADER_SIZE, 24);
667
668        // key map: 4 * (1 byte) = 4 bytes
669        // child map: 4 * (8 bytes (on 64-bit platform)) = 32
670        //
671        // 4 bytes of padding are inserted after the `keys` field to align the field to
672        // an 8 byte boundary.
673        assert_eq!(
674            mem::size_of::<InnerNode4<Box<[u8]>, usize, DEFAULT_PREFIX_LEN>>(),
675            EXPECTED_HEADER_SIZE + 40
676        );
677        // key map: 16 * (1 byte) = 16 bytes
678        // child map: 16 * (8 bytes (on 64-bit platform)) = 128
679        assert_eq!(
680            mem::size_of::<InnerNode16<Box<[u8]>, usize, DEFAULT_PREFIX_LEN>>(),
681            EXPECTED_HEADER_SIZE + 144
682        );
683        // key map: 256 * (1 byte) = 256 bytes
684        // child map: 48 * (8 bytes (on 64-bit platform)) = 384
685        assert_eq!(
686            mem::size_of::<InnerNode48<Box<[u8]>, usize, DEFAULT_PREFIX_LEN>>(),
687            EXPECTED_HEADER_SIZE + 640
688        );
689        // child & key map: 256 * (8 bytes (on 64-bit platform)) = 2048
690        assert_eq!(
691            mem::size_of::<InnerNodeDirect<Box<[u8]>, usize, DEFAULT_PREFIX_LEN>>(),
692            EXPECTED_HEADER_SIZE + 2048
693        );
694
695        // Assert that pointer is expected size and has non-null optimization
696        assert_eq!(
697            mem::size_of::<Option<OpaqueNodePtr<Box<[u8]>, (), DEFAULT_PREFIX_LEN>>>(),
698            8
699        );
700        assert_eq!(
701            mem::size_of::<OpaqueNodePtr<Box<[u8]>, (), DEFAULT_PREFIX_LEN>>(),
702            8
703        );
704    }

707    fn node_alignment() {
708        assert_eq!(mem::align_of::<InnerNode4<Box<[u8]>, u8, 16>>(), 8);
709        assert_eq!(mem::align_of::<InnerNode16<Box<[u8]>, u8, 16>>(), 8);
710        assert_eq!(mem::align_of::<InnerNode48<Box<[u8]>, u8, 16>>(), 8);
711        assert_eq!(mem::align_of::<InnerNodeDirect<Box<[u8]>, u8, 16>>(), 8);
712        assert_eq!(mem::align_of::<LeafNode<Box<[u8]>, u8, 16>>(), 8);
713        assert_eq!(mem::align_of::<Header<16>>(), 4);
714
715        assert_eq!(
716            mem::align_of::<InnerNode4<Box<[u8]>, u8, 16>>(),
717            mem::align_of::<OpaqueValue>()
718        );
719        assert_eq!(
720            mem::align_of::<InnerNode16<Box<[u8]>, u8, 16>>(),
721            mem::align_of::<OpaqueValue>()
722        );
723        assert_eq!(
724            mem::align_of::<InnerNode48<Box<[u8]>, u8, 16>>(),
725            mem::align_of::<OpaqueValue>()
726        );
727        assert_eq!(
728            mem::align_of::<InnerNodeDirect<Box<[u8]>, u8, 16>>(),
729            mem::align_of::<OpaqueValue>()
730        );
731        assert_eq!(
732            mem::align_of::<LeafNode<Box<[u8]>, u8, 16>>(),
733            mem::align_of::<OpaqueValue>()
734        );
735
736        let n4 = InnerNode4::<Box<[u8]>, (), 16>::empty();
737        let n16 = InnerNode4::<Box<[u8]>, (), 16>::empty();
738        let n48 = InnerNode4::<Box<[u8]>, (), 16>::empty();
739        let n256 = InnerNode4::<Box<[u8]>, (), 16>::empty();
740
741        let n4_ptr = const_addr(&n4 as *const InnerNode4<Box<[u8]>, (), 16>);
742        let n16_ptr = const_addr(&n16 as *const InnerNode4<Box<[u8]>, (), 16>);
743        let n48_ptr = const_addr(&n48 as *const InnerNode4<Box<[u8]>, (), 16>);
744        let n256_ptr = const_addr(&n256 as *const InnerNode4<Box<[u8]>, (), 16>);
745
746        // Ensure that there are 3 bits of unused space in the node pointer because of
747        // the alignment.
748        assert!(n4_ptr.trailing_zeros() >= 3);
749        assert!(n16_ptr.trailing_zeros() >= 3);
750        assert!(n48_ptr.trailing_zeros() >= 3);
751        assert!(n256_ptr.trailing_zeros() >= 3);
752    }

754    pub(crate) fn inner_node_write_child_test<const PREFIX_LEN: usize>(
755        mut node: impl InnerNode<PREFIX_LEN, Key = Box<[u8]>, Value = ()>,
756        num_children: usize,
757    ) {
758        let mut leaves = Vec::with_capacity(num_children);
759        for _ in 0..num_children {
760            leaves.push(LeafNode::with_no_siblings(vec![].into(), ()));
761        }
762
763        assert!(!node.is_full());
764        {
765            let leaf_pointers = leaves
766                .iter_mut()
767                .map(|leaf| NodePtr::from(leaf).to_opaque())
768                .collect::<Vec<_>>();
769
770            for (idx, leaf_pointer) in leaf_pointers.iter().copied().enumerate() {
771                node.write_child(u8::try_from(idx).unwrap(), leaf_pointer);
772            }
773
774            for (idx, leaf_pointer) in leaf_pointers.into_iter().enumerate() {
775                assert_eq!(
776                    node.lookup_child(u8::try_from(idx).unwrap()),
777                    Some(leaf_pointer)
778                );
779            }
780        }
781
782        assert!(node.is_full());
783    }

852    pub(crate) fn inner_node_min_max_test<const PREFIX_LEN: usize>(
853        mut node: impl InnerNode<PREFIX_LEN, Key = Box<[u8]>, Value = ()>,
854        num_children: usize,
855    ) {
856        assert!(
857            num_children >= 2,
858            "test harness must specify more than 1 child"
859        );
860
861        let mut leaves = Vec::with_capacity(num_children);
862        for _ in 0..num_children {
863            leaves.push(LeafNode::with_no_siblings(vec![].into(), ()));
864        }
865
866        let leaf_pointers = leaves
867            .iter_mut()
868            .map(|leaf| NodePtr::from(leaf).to_opaque())
869            .collect::<Vec<_>>();
870
871        for (idx, leaf_pointer) in leaf_pointers.iter().copied().enumerate() {
872            node.write_child(u8::try_from(idx).unwrap(), leaf_pointer);
873        }
874
875        assert_eq!(node.header().num_children(), num_children);
876
877        // generate list of interleaved `[(0, num_children - 1), (1, num_children -
878        // 2), (2, num_children - 3), ..., (midpoint, num_children - midpoint + 1)]`
879        let midpoint = num_children / 2;
880        let interleaved = (0..midpoint).zip((midpoint..num_children).rev());
881
882        for (min_idx, max_idx) in interleaved {
883            let (min_key_fragment, min_leaf_node) = node.min();
884            assert_eq!(min_leaf_node, node.remove_child(min_key_fragment).unwrap());
885            let expected_min_leaf_node =
886                NodePtr::from(leaves.get_mut(min_idx).unwrap()).to_opaque();
887            assert_eq!(expected_min_leaf_node, min_leaf_node);
888
889            let (max_key_fragment, max_leaf_node) = node.max();
890            assert_eq!(max_leaf_node, node.remove_child(max_key_fragment).unwrap());
891            let expected_max_leaf_node =
892                NodePtr::from(leaves.get_mut(max_idx).unwrap()).to_opaque();
893            assert_eq!(expected_max_leaf_node, max_leaf_node);
894        }
895
896        assert_eq!(node.header().num_children(), 0);
897    }
