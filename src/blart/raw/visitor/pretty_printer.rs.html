14/// Settings which customize the output of the [`DotPrinter`] visitor.
15#[derive(Debug, Default, Clone, PartialEq, Eq)]
16#[non_exhaustive]
17pub struct DotPrinterSettings {
18    /// Add node address to output in graphs
19    pub display_node_address: bool,
20}
21
22/// A visitor of the radix trie that will print the tree in "dot" notation.
23///
24/// See ['DOT Language | Graphviz'](https://graphviz.org/doc/info/lang.html) for
25/// information about syntax and example of the language.
26pub struct DotPrinter<O: Write> {
27    output: O,
28    next_id: usize,
29    settings: DotPrinterSettings,
30}
31
32impl<O: Write> DotPrinter<O> {
33    /// Write the dot-format of the given tree to the given output.
34    pub fn print<K, V, const PREFIX_LEN: usize, A: Allocator>(
35        output: O,
36        tree: &TreeMap<K, V, PREFIX_LEN, A>,
37        settings: DotPrinterSettings,
38    ) -> Option<io::Result<()>>
39    where
40        K: Display,
41        V: Display,
42    {
43        tree.state.as_ref().map(|state| {
44            // SAFETY: Since we get a reference to the `TreeMap`, we know the
45            // node and all descendants will not be mutated
46            unsafe { Self::print_tree(output, &state.root, settings) }
47        })
48    }
49
50    /// Write the dot-format of the given tree to the given output.
51    ///
52    /// # Safety
53    ///  - For the duration of this function, the given node and all its
54    ///    children nodes must not get mutated.
55    pub(crate) unsafe fn print_tree<K, V, const PREFIX_LEN: usize>(
56        output: O,
57        tree: &OpaqueNodePtr<K, V, PREFIX_LEN>,
58        settings: DotPrinterSettings,
59    ) -> io::Result<()>
60    where
61        K: Display,
62        V: Display,
63    {
64        let mut visitor = DotPrinter {
65            output,
66            next_id: 0,
67            settings,
68        };
69
70        visitor.output_prelude()?;
71        let _ = tree.visit_with(&mut visitor)?;
72        visitor.output_epilogue()
73    }
74
75    fn output_prelude(&mut self) -> io::Result<()> {
76        writeln!(self.output, "strict digraph G {{")?;
77        writeln!(self.output, "node [shape=record]")
78    }
79
80    fn output_epilogue(&mut self) -> io::Result<()> {
81        writeln!(self.output, "}}")
82    }
83
84    fn get_id(&mut self) -> usize {
85        let new_id = self.next_id;
86        self.next_id += 1;
87        new_id
88    }
89
90    fn write_inner_node<K, T, N, const PREFIX_LEN: usize>(
91        &mut self,
92        inner_node: &N,
93    ) -> io::Result<usize>
94    where
95        K: Display,
96        T: Display,
97        N: InnerNode<PREFIX_LEN, Key = K, Value = T>,
98    {
99        let header = inner_node.header();
100        let node_id = self.get_id();
101        write!(self.output, "n{node_id} ")?;
102        write!(self.output, "[label=\"{{")?;
103        // write header line
104        if self.settings.display_node_address {
105            write!(
106                self.output,
107                "{{<h0> {:p}}}  | {{{:?} | {:?} | {:?}}} | {{",
108                inner_node as *const _,
109                N::TYPE,
110                header.prefix_len(),
111                header.read_prefix()
112            )?;
113        } else {
114            write!(
115                self.output,
116                "{{<h0> {:?} | {:?} | {:?}}} | {{",
117                N::TYPE,
118                header.prefix_len(),
119                header.read_prefix()
120            )?;
121        }
122
123        // SAFETY: The `child_it` does not live beyond the following loop and will not
124        // overlap with any mutating access or operation, which is guaranteed by the
125        // `print_tree` caller requirements.
126        let child_it = inner_node.iter();
127        for (idx, (key_fragment, _)) in child_it.enumerate() {
128            if idx == 0 {
129                write!(self.output, "<c{idx}> {key_fragment}")?;
130            } else {
131                write!(self.output, "| <c{idx}> {key_fragment}")?;
132            }
133        }
134        writeln!(self.output, "}}}}\"]")?;
135
136        // SAFETY: The `child_it` does not live beyond the following loop and will not
137        // overlap with any mutating access or operation, which is guaranteed by the
138        // `print_tree` caller requirements.
139        let child_it = inner_node.iter();
140        for (key_frag_id, (_, child)) in child_it.enumerate() {
141            let child_id = child.visit_with(self)?;
142
143            writeln!(self.output, "n{node_id}:c{key_frag_id} -> n{child_id}:h0")?;
144        }
145
146        Ok(node_id)
147    }
148}

150impl<K, T, O, const PREFIX_LEN: usize> Visitor<K, T, PREFIX_LEN> for DotPrinter<O>
151where
152    K: Display,
153    T: Display,
154    O: Write,
155{
156    type Output = io::Result<usize>;
157
158    fn default_output(&self) -> Self::Output {
159        unimplemented!("this visitor should never use the default output")
160    }
161
162    fn combine_output(&self, _: Self::Output, _: Self::Output) -> Self::Output {
163        unimplemented!("this visitor should never combine outputs")
164    }
165
166    fn visit_node4(&mut self, t: &super::InnerNode4<K, T, PREFIX_LEN>) -> Self::Output {
167        self.write_inner_node(t)
168    }
169
170    fn visit_node16(&mut self, t: &super::InnerNode16<K, T, PREFIX_LEN>) -> Self::Output {
171        self.write_inner_node(t)
172    }
173
174    fn visit_node48(&mut self, t: &super::InnerNode48<K, T, PREFIX_LEN>) -> Self::Output {
175        self.write_inner_node(t)
176    }
177
178    fn visit_node256(&mut self, t: &super::InnerNode256<K, T, PREFIX_LEN>) -> Self::Output {
179        self.write_inner_node(t)
180    }
181
182    fn visit_leaf(&mut self, t: &super::LeafNode<K, T, PREFIX_LEN>) -> Self::Output {
183        let node_id = self.get_id();
184        write!(self.output, "n{node_id} ")?;
185        write!(self.output, "[label=\"{{")?;
186        // write header line
187        if self.settings.display_node_address {
188            writeln!(
189                self.output,
190                "{{<h0> {:p}}} | {{{:?}}} | {{{}}} | {{{}}} | {{{:p} {:p}}} }}\"]",
191                t as *const _,
192                NodeType::Leaf,
193                t.key_ref(),
194                t.value_ref(),
195                t.previous
196                    .map_or(std::ptr::null_mut(), |prev| prev.to_ptr()),
197                t.next.map_or(std::ptr::null_mut(), |next| next.to_ptr()),
198            )?;
199        } else {
200            writeln!(
201                self.output,
202                "{{<h0> {:?}}} | {{{}}} | {{{}}}}}\"]",
203                NodeType::Leaf,
204                t.key_ref(),
205                t.value_ref()
206            )?;
207        }
208
209        Ok(node_id)
210    }
211}

213/// This struct is intended to wrap a type which implements [`Debug`] and expose
214/// it as an implementation of [`Display`].
215///
216/// This is helpful when working with [`TreeMap`] key types which don't directly
217/// implement [`Display`], but you still want to print the tree using
218/// [`DotPrinter`].
219#[derive(Debug, Default, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
220pub struct DebugAsDisplay<T>(T);
221
222impl<T> Deref for DebugAsDisplay<T> {
223    type Target = T;
224
225    fn deref(&self) -> &Self::Target {
226        &self.0
227    }
228}
229
230impl<T: AsBytes> AsBytes for DebugAsDisplay<T> {
231    fn as_bytes(&self) -> &[u8] {
232        self.0.as_bytes()
233    }
234}
235
236// SAFETY: The underlying type already implements the `NoPrefixesBytes`, so this
237// is safe
238unsafe impl<T: NoPrefixesBytes> NoPrefixesBytes for DebugAsDisplay<T> {}
239
240// SAFETY: The underlying type already implements the `OrderedBytes`, so this is
241// safe
242unsafe impl<T: OrderedBytes> OrderedBytes for DebugAsDisplay<T> {}
243
244impl<T> Borrow<T> for DebugAsDisplay<T> {
245    fn borrow(&self) -> &T {
246        &self.0
247    }
248}
249
250impl<T> DebugAsDisplay<T> {
251    /// Consume this struct and return the inner value.
252    pub fn into_inner(self) -> T {
253        self.0
254    }
255}
256
257impl<T> From<T> for DebugAsDisplay<T> {
258    fn from(value: T) -> Self {
259        Self(value)
260    }
261}
262
263impl<T> fmt::Display for DebugAsDisplay<T>
264where
265    T: fmt::Debug,
266{
267    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
268        <T as fmt::Debug>::fmt(&self.0, f)
269    }
270}
