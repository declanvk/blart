14/// Settings which customize the output of the [`DotPrinter`] visitor.
15#[derive(Debug, Default, Clone, PartialEq, Eq)]
16#[non_exhaustive]
17pub struct DotPrinterSettings {
18    /// Add node address to output in graphs
19    pub display_node_address: bool,
20}

22/// A visitor of the radix trie that will print the tree in "dot" notation.
23///
24/// See ['DOT Language | Graphviz'](https://graphviz.org/doc/info/lang.html) for
25/// information about syntax and example of the language.
26#[derive(Debug)]
27pub struct DotPrinter<O: Write, K, V> {
28    output: O,
29    next_id: usize,
30    settings: DotPrinterSettings,
31    key_fmt: fn(&K, &mut fmt::Formatter<'_>) -> fmt::Result,
32    value_fmt: fn(&V, &mut fmt::Formatter<'_>) -> fmt::Result,
33}
34
35impl<O: Write, K, V> DotPrinter<O, K, V> {
36    /// Write the dot-format of the given tree to the given output.
37    pub fn print<const PREFIX_LEN: usize, A: Allocator>(
38        output: O,
39        tree: &TreeMap<K, V, PREFIX_LEN, A>,
40        settings: DotPrinterSettings,
41    ) -> Option<io::Result<()>>
42    where
43        K: fmt::Display,
44        V: fmt::Display,
45    {
46        Self::print_with_fmt(
47            output,
48            tree,
49            settings,
50            <K as fmt::Display>::fmt,
51            <V as fmt::Display>::fmt,
52        )
53    }
54
55    /// Write the dot-format of the given tree to the given output.
56    pub fn print_with_fmt<const PREFIX_LEN: usize, A: Allocator>(
57        output: O,
58        tree: &TreeMap<K, V, PREFIX_LEN, A>,
59        settings: DotPrinterSettings,
60        key_fmt: fn(&K, &mut fmt::Formatter<'_>) -> fmt::Result,
61        value_fmt: fn(&V, &mut fmt::Formatter<'_>) -> fmt::Result,
62    ) -> Option<io::Result<()>> {
63        tree.state.as_ref().map(|state| {
64            // SAFETY: Since we get a reference to the `TreeMap`, we know the
65            // node and all descendants will not be mutated
66            unsafe { Self::print_tree(output, &state.root, settings, key_fmt, value_fmt) }
67        })
68    }
69
70    /// Write the dot-format of the given tree to the given output.
71    ///
72    /// # Safety
73    ///  - For the duration of this function, the given node and all its
74    ///    children nodes must not get mutated.
75    pub(crate) unsafe fn print_tree<const PREFIX_LEN: usize>(
76        output: O,
77        tree: &OpaqueNodePtr<K, V, PREFIX_LEN>,
78        settings: DotPrinterSettings,
79        key_fmt: fn(&K, &mut fmt::Formatter<'_>) -> fmt::Result,
80        value_fmt: fn(&V, &mut fmt::Formatter<'_>) -> fmt::Result,
81    ) -> io::Result<()> {
82        let mut visitor = DotPrinter {
83            output,
84            next_id: 0,
85            settings,
86            key_fmt,
87            value_fmt,
88        };
89
90        visitor.output_prelude()?;
91        let _ = tree.visit_with(&mut visitor)?;
92        visitor.output_epilogue()
93    }
94
95    fn output_prelude(&mut self) -> io::Result<()> {
96        writeln!(self.output, "strict digraph G {{")?;
97        writeln!(self.output, "node [shape=record]")
98    }
99
100    fn output_epilogue(&mut self) -> io::Result<()> {
101        writeln!(self.output, "}}")
102    }
103
104    fn get_id(&mut self) -> usize {
105        let new_id = self.next_id;
106        self.next_id += 1;
107        new_id
108    }
109
110    fn write_inner_node<N, const PREFIX_LEN: usize>(&mut self, inner_node: &N) -> io::Result<usize>
111    where
112        N: InnerNode<PREFIX_LEN, Key = K, Value = V>,
113    {
114        let header = inner_node.header();
115        let node_id = self.get_id();
116        write!(self.output, "n{node_id} ")?;
117        write!(self.output, "[label=\"{{")?;
118        // write header line
119        if self.settings.display_node_address {
120            write!(
121                self.output,
122                "{{<h0> {:p}}}  | {{{:?} | {:?} | {:?}}} | {{",
123                ptr::from_ref(inner_node),
124                N::TYPE,
125                header.prefix_len(),
126                header.read_prefix()
127            )?;
128        } else {
129            write!(
130                self.output,
131                "{{<h0> {:?} | {:?} | {:?}}} | {{",
132                N::TYPE,
133                header.prefix_len(),
134                header.read_prefix()
135            )?;
136        }
137
138        // SAFETY: The `child_it` does not live beyond the following loop and will not
139        // overlap with any mutating access or operation, which is guaranteed by the
140        // `print_tree` caller requirements.
141        let child_it = inner_node.iter();
142        for (idx, (key_fragment, _)) in child_it.enumerate() {
143            if idx == 0 {
144                write!(self.output, "<c{idx}> {key_fragment}")?;
145            } else {
146                write!(self.output, "| <c{idx}> {key_fragment}")?;
147            }
148        }
149        writeln!(self.output, "}}}}\"]")?;
150
151        // SAFETY: The `child_it` does not live beyond the following loop and will not
152        // overlap with any mutating access or operation, which is guaranteed by the
153        // `print_tree` caller requirements.
154        let child_it = inner_node.iter();
155        for (key_frag_id, (_, child)) in child_it.enumerate() {
156            let child_id = child.visit_with(self)?;
157
158            writeln!(self.output, "n{node_id}:c{key_frag_id} -> n{child_id}:h0")?;
159        }
160
161        Ok(node_id)
162    }
163}

165impl<K, V, O, const PREFIX_LEN: usize> Visitor<K, V, PREFIX_LEN> for DotPrinter<O, K, V>
166where
167    O: Write,
168{
169    type Output = io::Result<usize>;
170
171    fn default_output(&self) -> Self::Output {
172        unimplemented!("this visitor should never use the default output")
173    }
174
175    fn combine_output(&self, _: Self::Output, _: Self::Output) -> Self::Output {
176        unimplemented!("this visitor should never combine outputs")
177    }
178
179    fn visit_inner_node<N>(&mut self, t: &N) -> Self::Output
180    where
181        N: InnerNode<PREFIX_LEN, Key = K, Value = V> + Visitable<K, V, PREFIX_LEN>,
182    {
183        self.write_inner_node(t)
184    }
185
186    fn visit_leaf(&mut self, t: &super::LeafNode<K, V, PREFIX_LEN>) -> Self::Output {
187        let key = OverrideDisplay {
188            value: t.key_ref(),
189            fmt_fn: self.key_fmt,
190        };
191        let value = OverrideDisplay {
192            value: t.value_ref(),
193            fmt_fn: self.value_fmt,
194        };
195
196        let node_id = self.get_id();
197        write!(self.output, "n{node_id} ")?;
198        write!(self.output, "[label=\"{{")?;
199        // write header line
200        if self.settings.display_node_address {
201            writeln!(
202                self.output,
203                "{{<h0> {:p}}} | {{{:?}}} | {{{}}} | {{{}}} | {{{:p} {:p}}} }}\"]",
204                ptr::from_ref(t),
205                NodeType::Leaf,
206                key,
207                value,
208                t.previous.map_or(ptr::null_mut(), NodePtr::to_ptr),
209                t.next.map_or(ptr::null_mut(), NodePtr::to_ptr),
210            )?;
211        } else {
212            writeln!(
213                self.output,
214                "{{<h0> {:?}}} | {{{}}} | {{{}}}}}\"]",
215                NodeType::Leaf,
216                key,
217                value
218            )?;
219        }
220
221        Ok(node_id)
222    }
223}

225#[derive(Debug)]
226struct OverrideDisplay<'a, T> {
227    value: &'a T,
228    fmt_fn: fn(&T, &mut fmt::Formatter<'_>) -> fmt::Result,
229}
230
231impl<T> fmt::Display for OverrideDisplay<'_, T> {
232    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
233        (self.fmt_fn)(self.value, f)
234    }
235}

237/// Print a given value using the [`Debug::fmt`] implementation.
238///
239/// This function can be used to override the formatting of key or value types
240/// when printing with [`DotPrinter`].
241///
242/// # Errors
243///
244/// This function returns an error if the provided [`fmt::Formatter`] returns
245/// `Err`.
246///
247/// [`Debug::fmt`]: std::fmt::Debug::fmt
248pub fn debug_as_display_fmt(value: &impl fmt::Debug, f: &mut fmt::Formatter) -> fmt::Result {
249    value.fmt(f)
250}
251
252/// Print the string "\[null\]".
253///
254/// This function can be used to override the formatting of key or value types
255/// when printing with [`DotPrinter`].
256///
257/// # Errors
258///
259/// This function returns an error if the provided [`fmt::Formatter`] returns
260/// `Err`.
261pub fn null_display_fmt(_value: &impl fmt::Debug, f: &mut fmt::Formatter) -> fmt::Result {
262    write!(f, "[null]")
263}
264
265/// Print the byte representation of the given value as returned by [`AsBytes`].
266///
267/// This function can be used to override the formatting of key or value types
268/// when printing with [`DotPrinter`].
269///
270/// # Errors
271///
272/// This function returns an error if the provided [`fmt::Formatter`] returns
273/// `Err`.
274pub fn bytes_display_fmt(value: &impl AsBytes, f: &mut fmt::Formatter) -> fmt::Result {
275    write!(f, "{:?}", value.as_bytes())
276}
