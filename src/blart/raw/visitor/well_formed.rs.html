19/// A portion of an entire key that should uniquely identify each node in
20/// the tree.
21///
22/// We assume that this should be unique for each node given no loops in the
23/// tree.
24#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Default, Hash)]
25pub struct KeyPrefix(Box<[u8]>);

39/// An issue with the well-formed-ness of the tree. See the documentation on
40/// [`WellFormedChecker`] for more context.
41#[derive(PartialEq, Eq)]
42#[non_exhaustive]
43pub enum MalformedTreeError<K, V, const PREFIX_LEN: usize> {
44    /// A loop was observed between nodes
45    LoopFound {
46        /// The node that was observed more than once while traversing the tree
47        node_ptr: OpaqueNodePtr<K, V, PREFIX_LEN>,
48        /// The key prefix when the node was first observed
49        first_observed: KeyPrefix,
50        /// The key prefix when the node was observed a second time
51        later_observed: KeyPrefix,
52    },
53    /// An inner node had an incorrect number of children
54    WrongChildrenCount {
55        /// The key prefix identifying the inner node
56        key_prefix: KeyPrefix,
57        /// The type of the inner node (InnerNode4, InnerNode16, etc)
58        ///
59        /// This field is guaranteed not to be [`NodeType::Leaf`]
60        inner_node_type: NodeType,
61        /// The number of children found at the inner node
62        num_children: usize,
63    },
64    /// The expected key prefix did not match the actual prefix that was present
65    /// in the leaf
66    PrefixMismatch {
67        /// The expected key prefix
68        expected_prefix: KeyPrefix,
69        /// The entire key
70        entire_key: Vec<u8>,
71    },
72    /// The length of the tree is not 0, even though the root is
73    /// [`Option::None`]
74    EmptyTreeWithLen,
75    /// There is a leaf node with a sibling pointer where the sibling has the
76    /// wrong key ordering with respect to the original leaf node.
77    LeafSiblingWrongOrder {
78        /// The key bytes of the sibling leaf node
79        sibling_key: Vec<u8>,
80        /// The key bytes of the leaf node that was out of order
81        leaf_key: Vec<u8>,
82        /// The expected ordering (either [`Ordering::Greater`] or
83        /// [`Ordering::Less`]) of the sibling and leaf keys.
84        expected_order: Ordering,
85        /// The actual ordering (either [`Ordering::Greater`] or
86        /// [`Ordering::Less`]) of the sibling and leaf keys.
87        actual_order: Ordering,
88    },
89    /// There is a leaf node which has incorrect values for either the
90    /// `previous` or `next` sibling pointers.
91    WrongSiblingLinks {
92        /// The key bytes of the broken leaf node
93        leaf_key: Vec<u8>,
94        /// The expected `previous` pointer value
95        expected_previous: OptionalLeafPtr<K, V, PREFIX_LEN>,
96        /// The expected `next` pointer value
97        expected_next: OptionalLeafPtr<K, V, PREFIX_LEN>,
98        /// The actual `previous` pointer value
99        actual_previous: OptionalLeafPtr<K, V, PREFIX_LEN>,
100        /// The actual `next` pointer value
101        actual_next: OptionalLeafPtr<K, V, PREFIX_LEN>,
102    },
103}

170impl<K, V, const PREFIX_LEN: usize> fmt::Display for MalformedTreeError<K, V, PREFIX_LEN>
171where
172    K: AsBytes,
173{
174    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
175        match self {
176            MalformedTreeError::LoopFound {
177                node_ptr,
178                first_observed,
179                later_observed,
180            } => {
181                write!(
182                    f,
183                    "Found a loop in the tree containing the node [{node_ptr:?}]. First observed \
184                     that node at [{first_observed:?}], then later observed the same node at \
185                     [{later_observed:?}]",
186                )
187            },
188            MalformedTreeError::WrongChildrenCount {
189                key_prefix,
190                inner_node_type,
191                num_children,
192            } => {
193                write!(
194                    f,
195                    "Found an inner node of type [{inner_node_type:?}] at location \
196                     [{key_prefix:?}] that had the wrong number of children! Expected children in \
197                     range [{:?}], but found [{num_children}] children",
198                    inner_node_type.capacity_range(),
199                )
200            },
201            MalformedTreeError::PrefixMismatch {
202                expected_prefix,
203                entire_key,
204            } => {
205                write!(
206                    f,
207                    "Found a leaf that had a mismatched key from the expected prefix! Expected \
208                     the leaf key to start with [{expected_prefix:?}], but the leaf key was [{:?}]",
209                    entire_key.as_bytes()
210                )
211            },
212            MalformedTreeError::EmptyTreeWithLen => {
213                write!(
214                    f,
215                    "The length of the tree is not 0, even though the root is None",
216                )
217            },
218            MalformedTreeError::LeafSiblingWrongOrder {
219                sibling_key,
220                leaf_key,
221                expected_order,
222                actual_order,
223            } => {
224                write!(
225                    f,
226                    "Found a leaf with key [{leaf_key:?}] and sibling with key [{sibling_key:?}] \
227                     that was expected to have [{expected_order:?}] but had [{actual_order:?}]"
228                )
229            },
230            MalformedTreeError::WrongSiblingLinks {
231                leaf_key,
232                expected_previous,
233                expected_next,
234                actual_previous,
235                actual_next,
236            } => {
237                if expected_previous != actual_previous && expected_next != actual_next {
238                    write!(
239                        f,
240                        "Found a leaf with key [{leaf_key:?}] where the previous pointer expected \
241                         to be [{expected_previous:?}] but was actually [{actual_previous:?}] and \
242                         the next pointer expected to be [{expected_next:?}] but was actually \
243                         [{actual_next:?}]."
244                    )
245                } else if expected_previous != actual_previous {
246                    write!(
247                        f,
248                        "Found a leaf with key [{leaf_key:?}] where the previous pointer expected \
249                         to be [{expected_previous:?}] but was actually [{actual_previous:?}]."
250                    )
251                } else {
252                    write!(
253                        f,
254                        "Found a leaf with key [{leaf_key:?}] where the next pointer expected to \
255                         be [{expected_next:?}] but was actually [{actual_next:?}]."
256                    )
257                }
258            },
259        }
260    }

322/// A visitor of the radix tree which checks that the tree is well-formed.
323///
324/// In this context, well-formed means that in the tree:
325///  1. there are no loops between nodes
326///  2. every inner node has a number of children that is in range for the inner
327///     node type. For example, InnerNode16 has between 5 and 16 children.
328///  3. the elements of the key (as part of inner node prefixes and child
329///     pointers) combine to match the leaf node key prefix
330///  4. the `previous` and `next` pointers that form a doubly-linked list of
331///     leaf nodes has no loops, and the ordering of the leaves in the list is
332///     equal to the ordering of the leaves when sorted by key. The linked list
333///     should also be properly terminated with `previous = None` at the start
334///     and `next = None` at the end.
335///
336/// #1 and #3 are unlikely, but #2 is a possibility if specific tree operations
337/// are not implemented correctly. This visitor can be used to sanity check the
338/// tree in unit tests or other test cases.
339///
340/// This checker will only return a single issue at a time. A tree is only
341/// "well-formed" (by the definition given above) if the checker returns
342/// `Ok(())`.
343#[derive(Debug)]
344pub struct WellFormedChecker<K, V, const PREFIX_LEN: usize> {
345    current_key_prefix: Vec<u8>,
346    seen_nodes: HashMap<OpaqueNodePtr<K, V, PREFIX_LEN>, KeyPrefix>,
347    seen_leaf_nodes: Vec<NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>>,
348}
349
350impl<K, V, const PREFIX_LEN: usize> WellFormedChecker<K, V, PREFIX_LEN>
351where
352    K: AsBytes,
353{
354    /// Traverse the given tree and check that it is well-formed. Returns the
355    /// number of leaf nodes in the tree.
356    ///
357    /// # Errors
358    ///  - Returns an error if the given tree is not well-formed.
359    pub fn check<A: Allocator>(
360        tree: &TreeMap<K, V, PREFIX_LEN, A>,
361    ) -> Result<WellFormedTreeStats, MalformedTreeError<K, V, PREFIX_LEN>> {
362        tree.state
363            .as_ref()
364            .map(|state| {
365                // SAFETY: Since we get a reference to the TreeMap, we know no
366                // mutation can happen to any of the nodes
367                unsafe { Self::check_tree(state.root) }
368            })
369            .unwrap_or_else(|| {
370                if tree.is_empty() {
371                    Ok(WellFormedTreeStats::default())
372                } else {
373                    Err(MalformedTreeError::EmptyTreeWithLen)
374                }
375            })
376    }
377
378    /// Traverse the given tree and check that it is well-formed. Returns the
379    /// number of leaf nodes in the tree.
380    ///
381    /// # Safety
382    ///  - For the duration of this function, the given node and all its
383    ///    children nodes must not be mutated.
384    ///
385    /// # Errors
386    ///  - Returns an error if the given tree is not well-formed.
387    pub(crate) unsafe fn check_tree(
388        tree: OpaqueNodePtr<K, V, PREFIX_LEN>,
389    ) -> Result<WellFormedTreeStats, MalformedTreeError<K, V, PREFIX_LEN>> {
390        let mut visitor = WellFormedChecker {
391            current_key_prefix: vec![],
392            seen_nodes: HashMap::new(),
393            seen_leaf_nodes: vec![],
394        };
395
396        // We see the root node at the empty prefix
397        visitor.seen_nodes.insert(tree, KeyPrefix::default());
398        if let Some(leaf_ptr) = tree.cast::<LeafNode<K, V, PREFIX_LEN>>() {
399            visitor.seen_leaf_nodes.push(leaf_ptr);
400        }
401
402        let stats = tree.visit_with(&mut visitor)?;
403
404        debug_assert_eq!(stats.num_leaf, visitor.seen_leaf_nodes.len());
405
406        visitor.verify_leaves_linked_list()?;
407
408        Ok(stats)
409    }
410
411    fn verify_leaves_linked_list(&self) -> Result<(), MalformedTreeError<K, V, PREFIX_LEN>> {
412        for (idx, leaf_ptr) in self.seen_leaf_nodes.iter().enumerate() {
413            let leaf = leaf_ptr.read();
414
415            let expected_previous = if idx == 0 {
416                None
417            } else {
418                Some(self.seen_leaf_nodes[idx - 1])
419            };
420
421            let expected_next = if idx == self.seen_leaf_nodes.len() - 1 {
422                None
423            } else {
424                Some(self.seen_leaf_nodes[idx + 1])
425            };
426
427            if leaf.previous != expected_previous || leaf.next != expected_next {
428                return Err(MalformedTreeError::WrongSiblingLinks {
429                    leaf_key: leaf.key_ref().as_bytes().to_vec(),
430                    expected_previous,
431                    expected_next,
432                    actual_previous: leaf.previous,
433                    actual_next: leaf.next,
434                });
435            }
436        }
437
438        Ok(())
439    }
440
441    fn visit_inner_node<N>(
442        &mut self,
443        inner_node: &N,
444    ) -> Result<WellFormedTreeStats, MalformedTreeError<K, V, PREFIX_LEN>>
445    where
446        N: InnerNode<PREFIX_LEN, Key = K, Value = V>,
447    {
448        let original_key_prefix_len = self.current_key_prefix.len();
449
450        // update running key prefix with inner node partial prefix
451        // TODO: Fix this, here the we should return the full reconstructed prefix if
452        // prefix len > PREFIX_LEN
453        self.current_key_prefix
454            .extend(inner_node.read_full_prefix(original_key_prefix_len).0);
455
456        // SAFETY: The `child_it` does not live beyond the following loop and will not
457        // overlap with any mutating access or operation, which is guaranteed by the
458        // `check_tree` caller requirements.
459        let child_it = inner_node.iter();
460
461        let mut running_node_count = WellFormedTreeStats::default();
462        let mut num_children: usize = 0;
463        for (key_byte, child_pointer) in child_it {
464            // update running key prefix with child pointer key fragment
465            self.current_key_prefix.push(key_byte);
466
467            let current_key_prefix: KeyPrefix = self.current_key_prefix.as_slice().into();
468
469            match self.seen_nodes.entry(child_pointer) {
470                Entry::Occupied(entry) => {
471                    return Err(MalformedTreeError::LoopFound {
472                        node_ptr: child_pointer,
473                        first_observed: entry.get().clone(),
474                        later_observed: current_key_prefix,
475                    });
476                },
477                Entry::Vacant(entry) => {
478                    entry.insert(current_key_prefix);
479                },
480            }
481
482            if let Some(leaf_node_ptr) = child_pointer.cast::<LeafNode<K, V, PREFIX_LEN>>() {
483                self.seen_leaf_nodes.push(leaf_node_ptr);
484            }
485
486            running_node_count += child_pointer.visit_with(self)?;
487
488            // remove child pointer key fragment
489            assert_eq!(
490                self.current_key_prefix
491                    .pop()
492                    .expect("should match push of key byte"),
493                key_byte
494            );
495
496            num_children += 1;
497        }
498
499        // remove inner node partial key prefix
500        self.current_key_prefix.truncate(original_key_prefix_len);
501
502        if !(N::TYPE.capacity_range().contains(&num_children)) {
503            let current_key_prefix: KeyPrefix = self.current_key_prefix.as_slice().into();
504            return Err(MalformedTreeError::WrongChildrenCount {
505                key_prefix: current_key_prefix,
506                inner_node_type: N::TYPE,
507                num_children,
508            });
509        }
510
511        running_node_count.num_inner += 1;
512
513        Ok(running_node_count)
514    }
515}

517/// This struct contains some simple stats collected from the trie when visiting
518/// it with [`WellFormedChecker`].
519#[derive(Debug, Default)]
520pub struct WellFormedTreeStats {
521    /// The number of leaf nodes in the trie.
522    pub num_leaf: usize,
523    /// The number of inner node in the trie.
524    pub num_inner: usize,
525}
526
527impl WellFormedTreeStats {
528    /// The total number of leaf nodes and inner nodes in the trie.
529    pub fn total_nodes(self) -> usize {
530        self.num_inner + self.num_leaf
531    }
532}
533
534impl Add for WellFormedTreeStats {
535    type Output = Self;
536
537    fn add(self, rhs: Self) -> Self::Output {
538        Self {
539            num_leaf: self.num_leaf + rhs.num_leaf,
540            num_inner: self.num_inner + rhs.num_inner,

542    }
543}
544
545impl AddAssign for WellFormedTreeStats {
546    fn add_assign(&mut self, rhs: Self) {
547        self.num_inner += rhs.num_inner;
548        self.num_leaf += rhs.num_leaf;
549    }

552impl<K, V, const PREFIX_LEN: usize> Visitor<K, V, PREFIX_LEN>
553    for WellFormedChecker<K, V, PREFIX_LEN>
554where
555    K: AsBytes,
556{
557    type Output = Result<WellFormedTreeStats, MalformedTreeError<K, V, PREFIX_LEN>>;
558
559    fn default_output(&self) -> Self::Output {
560        // Chose zero so that any places that call `default_output` don't influence the
561        // overall count
562        Ok(Default::default())
563    }
564
565    fn combine_output(&self, o1: Self::Output, o2: Self::Output) -> Self::Output {
566        Ok(o1? + o2?)
567    }
568
569    fn visit_node4(&mut self, t: &super::InnerNode4<K, V, PREFIX_LEN>) -> Self::Output {
570        self.visit_inner_node(t)
571    }
572
573    fn visit_node16(&mut self, t: &super::InnerNode16<K, V, PREFIX_LEN>) -> Self::Output {
574        self.visit_inner_node(t)
575    }
576
577    fn visit_node48(&mut self, t: &super::InnerNode48<K, V, PREFIX_LEN>) -> Self::Output {
578        self.visit_inner_node(t)
579    }
580
581    fn visit_node256(&mut self, t: &super::InnerNode256<K, V, PREFIX_LEN>) -> Self::Output {
582        self.visit_inner_node(t)
583    }
584
585    fn visit_leaf(&mut self, t: &super::LeafNode<K, V, PREFIX_LEN>) -> Self::Output {
586        if !t.key_ref().as_bytes().starts_with(&self.current_key_prefix) {
587            let current_key_prefix: KeyPrefix = self.current_key_prefix.as_slice().into();
588            return Err(MalformedTreeError::PrefixMismatch {
589                expected_prefix: current_key_prefix,
590                entire_key: t.key_ref().as_bytes().to_vec(),
591            });
592        }
593
594        if let Some(sibling_ptr) = t.previous {
595            let sibling = sibling_ptr.read();
596            let sibling_order = sibling.key_ref().as_bytes().cmp(t.key_ref().as_bytes());
597            if sibling_order != Ordering::Less {
598                return Err(MalformedTreeError::LeafSiblingWrongOrder {
599                    sibling_key: sibling.key_ref().as_bytes().to_vec(),
600                    leaf_key: t.key_ref().as_bytes().to_vec(),
601                    expected_order: Ordering::Less,
602                    actual_order: sibling_order,
603                });
604            }
605        }
606
607        if let Some(sibling_ptr) = t.next {
608            let sibling = sibling_ptr.read();
609            let sibling_order = sibling.key_ref().as_bytes().cmp(t.key_ref().as_bytes());
610            if sibling_order != Ordering::Greater {
611                return Err(MalformedTreeError::LeafSiblingWrongOrder {
612                    sibling_key: sibling.key_ref().as_bytes().to_vec(),
613                    leaf_key: t.key_ref().as_bytes().to_vec(),
614                    expected_order: Ordering::Greater,
615                    actual_order: sibling_order,
616                });
617            }
618        }
619
620        Ok(WellFormedTreeStats {
621            num_inner: 0,
622            num_leaf: 1,
623        })
624    }
625}

674    #[test]
675    fn check_tree_with_loop() {
676        // have to allocate in this one because miri didn't like us using `&mut _` to
677        // make loops
678
679        let l1 = LeafNode::with_no_siblings(Box::new([1, 2, 3, 5, 6, 1]), 123561);
680        let l2 = LeafNode::with_no_siblings(Box::new([1, 2, 3, 5, 6, 2]), 123562);
681        let l3 = LeafNode::with_no_siblings(Box::new([1, 2, 4, 7, 8, 3]), 124783);
682
683        let l1_ptr: NodePtr<16, LeafNode<Box<[u8; 6]>, i32, 16>> =
684            NodePtr::allocate_node_ptr(l1, &Global);
685        let l2_ptr = NodePtr::allocate_node_ptr(l2, &Global);
686        let l3_ptr = NodePtr::allocate_node_ptr(l3, &Global);
687
688        let n4_left = InnerNode4::from_prefix(&[5, 6], 2);
689        let n4_right = InnerNode4::from_prefix(&[7, 8], 2);
690        let n16 = InnerNode16::from_prefix(&[1, 2], 2);
691
692        let n4_left_ptr = NodePtr::allocate_node_ptr(n4_left, &Global);
693        let n4_right_ptr = NodePtr::allocate_node_ptr(n4_right, &Global);
694
695        // construct root early
696        let root = NodePtr::allocate_node_ptr(n16, &Global);
697
698        {
699            let n4_left = unsafe { n4_left_ptr.as_mut() };
700            // Update inner node prefix and child slots
701            n4_left.write_child(1, l1_ptr.to_opaque());
702            n4_left.write_child(2, l2_ptr.to_opaque());
703        }
704
705        {
706            let n4_right = unsafe { n4_right_ptr.as_mut() };
707
708            n4_right.write_child(3, l3_ptr.to_opaque());
709            // replace normal l4 pointer with loop back to root
710            n4_right.write_child(4, root.to_opaque());
711        }
712
713        {
714            let n16 = unsafe { root.as_mut() };
715            n16.write_child(3, n4_left_ptr.to_opaque());
716            n16.write_child(4, n4_right_ptr.to_opaque());
717        }
718
719        let check_result = unsafe { WellFormedChecker::check_tree(root.to_opaque()) }
720            .expect_err("should have failed for loop");
721        match check_result {
722            MalformedTreeError::LoopFound {
723                node_ptr,
724                first_observed,
725                later_observed,
726            } => {
727                assert_eq!(node_ptr, root.to_opaque());
728                assert_eq!(first_observed, []);
729                assert_eq!(later_observed, [1u8, 2, 4, 7, 8, 4]);
730            },
731            _ => {
732                panic!("expected a LoopFound error")
733            },
734        }
735
736        // We can't just call `deallocate_tree(root)` because the deallocate function
737        // assumes no loops, if we did use `deallocate_tree` it would hit a
738        // use-after-free error
739        unsafe {
740            let _ = NodePtr::deallocate_node_ptr(root, &Global);
741        };
742        unsafe {
743            let _ = NodePtr::deallocate_node_ptr(n4_left_ptr, &Global);
744        };
745        unsafe {
746            let _ = NodePtr::deallocate_node_ptr(n4_right_ptr, &Global);
747        };
748        unsafe {
749            let _ = NodePtr::deallocate_node_ptr(l1_ptr, &Global);
750        };
751        unsafe {
752            let _ = NodePtr::deallocate_node_ptr(l2_ptr, &Global);
753        };
754        unsafe {
755            let _ = NodePtr::deallocate_node_ptr(l3_ptr, &Global);
756        };
757    }
