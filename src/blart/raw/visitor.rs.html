20/// The `Visitable` trait allows [`Visitor`]s to traverse the structure of the
21/// implementing type and produce some output.
22pub trait Visitable<K, T, const PREFIX_LEN: usize> {
23    /// This function provides the default traversal behavior for the
24    /// implementing type.
25    ///
26    /// The implementation should call `visit_with(visitor)` for all relevant
27    /// sub-fields of the type. If there are no relevant sub-fields, it should
28    /// just produce the default output.
29    fn super_visit_with<V: Visitor<K, T, PREFIX_LEN>>(&self, visitor: &mut V) -> V::Output;
30
31    /// This function will traverse the implementing type and execute any
32    /// specific logic from the given [`Visitor`].
33    ///
34    /// This function should be override for types that have corresponding hooks
35    /// in the [`Visitor`] trait. For example the [`Visitable`] implementation
36    /// for [`InnerNode4`] looks like:
37    ///
38    /// ```rust,compile_fail
39    /// impl<K, T> Visitable<K, T, PREFIX_LEN> for InnerNode4<K, T> {
40    ///     ...
41    ///
42    ///     fn visit_with<V: Visitor<K, T, PREFIX_LEN>>(&self, visitor: &mut V) -> V::Output {
43    ///         visitor.visit_node4(self)
44    ///     }
45    /// }
46    /// ```
47    ///
48    /// The call to `visitor.visit_node4(self)` allows the visitor to execute
49    /// specific handling logic.
50    fn visit_with<V: Visitor<K, T, PREFIX_LEN>>(&self, visitor: &mut V) -> V::Output {
51        self.super_visit_with(visitor)
52    }
53}

55impl<K, T, const PREFIX_LEN: usize> Visitable<K, T, PREFIX_LEN>
56    for OpaqueNodePtr<K, T, PREFIX_LEN>
57{
58    fn super_visit_with<V: Visitor<K, T, PREFIX_LEN>>(&self, visitor: &mut V) -> V::Output {
59        match self.to_node_ptr() {
60            ConcreteNodePtr::Node4(inner) => inner.visit_with(visitor),
61            ConcreteNodePtr::Node16(inner) => inner.visit_with(visitor),
62            ConcreteNodePtr::Node48(inner) => inner.visit_with(visitor),
63            ConcreteNodePtr::Node256(inner) => inner.visit_with(visitor),
64            ConcreteNodePtr::LeafNode(inner) => inner.visit_with(visitor),
65        }
66    }
67}

69impl<K, T, const PREFIX_LEN: usize, N: Node<PREFIX_LEN> + Visitable<K, T, PREFIX_LEN>>
70    Visitable<K, T, PREFIX_LEN> for NodePtr<PREFIX_LEN, N>
71{
72    fn super_visit_with<V: Visitor<K, T, PREFIX_LEN>>(&self, visitor: &mut V) -> V::Output {
73        // FIXME(https://github.com/declanvk/blart/issues/34): This is broken in a couple ways:
74        //  1. The `Visitor`/`Visitable` trait methods likely all need to be marked
75        //     unsafe, since they are operating on the raw nodes of the tree. The
76        //     visitor implementations so far include "Safety" doc-comments requiring
77        //     read-only access, but it probably should be recorded in the function
78        //     signature
79        //  2. The `visit_with` functions work better when they're using a reference
80        //     pointing to the actual location of `N`, not a local copy on the stack.
81        //     For example, the DotPrinter will attempt to print node addresses by
82        //     converting the given reference into a pointer, but this only really works
83        //     if the reference points to the actual node location.
84        // let inner = self.read();
85        // inner.visit_with(visitor)
86        unsafe { self.as_ref().visit_with(visitor) }
87    }
88}

140/// The `Visitor` trait allows creating new operations on the radix tree by
141/// overriding specific handling methods for each of the node types.
142pub trait Visitor<K, V, const PREFIX_LEN: usize>: Sized {
143    /// The type of value that the visitor produces.
144    type Output;
145
146    /// Produce the default value of the [`Self::Output`] type.
147    fn default_output(&self) -> Self::Output;
148
149    /// Combine two instances of the [`Self::Output`] type for this [`Visitor`].
150    fn combine_output(&self, o1: Self::Output, o2: Self::Output) -> Self::Output;
151
152    /// Visit a [`InnerNode4`].
153    fn visit_node4(&mut self, t: &InnerNode4<K, V, PREFIX_LEN>) -> Self::Output {
154        t.super_visit_with(self)
155    }
156
157    /// Visit a [`InnerNode16`].
158    fn visit_node16(&mut self, t: &InnerNode16<K, V, PREFIX_LEN>) -> Self::Output {
159        t.super_visit_with(self)
160    }
161
162    /// Visit a [`InnerNode48`].
163    fn visit_node48(&mut self, t: &InnerNode48<K, V, PREFIX_LEN>) -> Self::Output {
164        t.super_visit_with(self)
165    }
166
167    /// Visit a [`InnerNode256`].
168    fn visit_node256(&mut self, t: &InnerNode256<K, V, PREFIX_LEN>) -> Self::Output {
169        t.super_visit_with(self)
170    }
171
172    /// Visit a [`LeafNode`].
173    fn visit_leaf(&mut self, t: &LeafNode<K, V, PREFIX_LEN>) -> Self::Output {
174        t.super_visit_with(self)
175    }
176}

178fn combine_inner_node_child_output<K, T, const PREFIX_LEN: usize, V: Visitor<K, T, PREFIX_LEN>>(
179    mut iter: impl Iterator<Item = (u8, OpaqueNodePtr<K, T, PREFIX_LEN>)>,
180    visitor: &mut V,
181) -> V::Output {
182    if let Some((_, first)) = iter.next() {
183        let mut accum = first.visit_with(visitor);
184        for (_, child) in iter {
185            let output = child.visit_with(visitor);
186            accum = visitor.combine_output(accum, output);
187        }
188
189        accum
190    } else {
191        visitor.default_output()
192    }
193}
