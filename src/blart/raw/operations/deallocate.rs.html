8/// Deallocate all the leaf nodes in the linked list starting that are within
9/// the given iterator.
10///
11/// # Safety
12///  - This function must only be called once for this `start` node and all
13///    following leaf nodes in the linked list, otherwise a double-free could
14///    result.
15///  - This function should not be called concurrently with any read of the
16///    tree, otherwise it could result in a use-after-free.
17///  - `alloc` must be the same allocator which was used to allocate all the
18///    leaves of the tree.
19pub unsafe fn deallocate_leaves<K, V, const PREFIX_LEN: usize, A: Allocator>(
20    mut leaf_range: RawIterator<K, V, PREFIX_LEN>,
21    alloc: &A,
22) {
23    // SAFETY: Covered by function safety doc
24    while let Some(leaf_ptr) = unsafe { leaf_range.next() } {
25        // SAFETY: Covered by function safety doc
26        let _ = unsafe { NodePtr::deallocate_node_ptr(leaf_ptr, alloc) };
27    }
28}

30/// Deallocate the given node and all children of the given node except for
31/// leaf nodes.
32///
33/// # Safety
34///  - This function must only be called once for this root node and all
35///    descendants, otherwise a double-free could result.
36///  - This function should not be called concurrently with any read of the
37///    tree, otherwise it could result in a use-after-free.
38///  - `alloc` must be the same allocator which was used to allocate all the
39///    inner nodes of the tree.
40pub unsafe fn deallocate_tree_non_leaves<K, V, const PREFIX_LEN: usize, A: Allocator>(
41    root: OpaqueNodePtr<K, V, PREFIX_LEN>,
42    alloc: &A,
43) {
44    fn not_leaf<K, V, const PREFIX_LEN: usize>(node: &OpaqueNodePtr<K, V, PREFIX_LEN>) -> bool {
45        !node.is::<LeafNode<K, V, PREFIX_LEN>>()
46    }
47
48    if root.is::<LeafNode<K, V, PREFIX_LEN>>() {
49        return;
50    }
51
52    let mut stack = Vec::new();
53
54    stack.push(root);
55
56    while let Some(next_node_ptr) = stack.pop() {
57        match next_node_ptr.to_node_ptr() {
58            ConcreteNodePtr::Node4(inner_ptr) => unsafe {
59                // SAFETY: Covered by function safety doc
60                deallocate_inner_node(&mut stack, inner_ptr, not_leaf, alloc)
61            },
62            ConcreteNodePtr::Node16(inner_ptr) => unsafe {
63                // SAFETY: Covered by function safety doc
64                deallocate_inner_node(&mut stack, inner_ptr, not_leaf, alloc)
65            },
66            ConcreteNodePtr::Node48(inner_ptr) => unsafe {
67                // SAFETY: Covered by function safety doc
68                deallocate_inner_node(&mut stack, inner_ptr, not_leaf, alloc)
69            },
70            ConcreteNodePtr::Node256(inner_ptr) => unsafe {
71                // SAFETY: Covered by function safety doc
72                deallocate_inner_node(&mut stack, inner_ptr, not_leaf, alloc)
73            },
74            ConcreteNodePtr::LeafNode(_) => {
75                unreachable!("should never encounter a leaf node")
76            },
77        }
78    }
79}

81/// Deallocate the given node and all children of the given node.
82///
83/// This will also deallocate the leaf nodes with their value type data.
84/// This function returns the amount of leaf nodes deallocated.
85///
86/// # Safety
87///  - This function must only be called once for this root node and all
88///    descendants, otherwise a double-free could result.
89///  - This function should not be called concurrently with any read of the
90///    tree, otherwise it could result in a use-after-free.
91///  - `alloc` must be the same allocator which was used to allocate all the
92///    nodes of the tree.
93pub unsafe fn deallocate_tree<K, V, const PREFIX_LEN: usize, A: Allocator>(
94    root: OpaqueNodePtr<K, V, PREFIX_LEN>,
95    alloc: &A,
96) -> usize {
97    fn accept_all<K, V, const PREFIX_LEN: usize>(_: &OpaqueNodePtr<K, V, PREFIX_LEN>) -> bool {
98        true
99    }
100
101    let mut count = 0;
102    let mut stack = Vec::new();
103
104    stack.push(root);
105
106    while let Some(next_node_ptr) = stack.pop() {
107        match next_node_ptr.to_node_ptr() {
108            ConcreteNodePtr::Node4(inner_ptr) => unsafe {
109                // SAFETY: Covered by function safety doc
110                deallocate_inner_node(&mut stack, inner_ptr, accept_all, alloc)
111            },
112            ConcreteNodePtr::Node16(inner_ptr) => unsafe {
113                // SAFETY: Covered by function safety doc
114                deallocate_inner_node(&mut stack, inner_ptr, accept_all, alloc)
115            },
116            ConcreteNodePtr::Node48(inner_ptr) => unsafe {
117                // SAFETY: Covered by function safety doc
118                deallocate_inner_node(&mut stack, inner_ptr, accept_all, alloc)
119            },
120            ConcreteNodePtr::Node256(inner_ptr) => unsafe {
121                // SAFETY: Covered by function safety doc
122                deallocate_inner_node(&mut stack, inner_ptr, accept_all, alloc)
123            },
124            ConcreteNodePtr::LeafNode(inner) => {
125                // SAFETY: The single call per node requirement is enforced by the safety
126                // requirements on this function.
127                drop(unsafe { NodePtr::deallocate_node_ptr(inner, alloc) });
128                count += 1;
129            },
130        }
131    }
132    count
133}

135/// This function will read an [`InnerNode`] and place all children that
136/// satisfy the `filter_children` predicate on the `stack`. Then it will
137/// deallocate the given [`InnerNode`].
138///
139/// # Safety
140///  - This function must only be called once for this inner node
141///  - This function cannot be called concurrently with any read of modification
142///    of the trie.
143unsafe fn deallocate_inner_node<K, V, N, A: Allocator, const PREFIX_LEN: usize>(
144    stack: &mut Vec<OpaqueNodePtr<K, V, PREFIX_LEN>>,
145    inner_ptr: NodePtr<PREFIX_LEN, N>,
146    filter_children: for<'a> fn(&'a OpaqueNodePtr<K, V, PREFIX_LEN>) -> bool,
147    alloc: &A,
148) where
149    N: InnerNode<PREFIX_LEN, Key = K, Value = V>,
150{
151    {
152        // SAFETY: The scope of this reference is bounded and we enforce that no
153        // mutation of the reference memory takes place within the lifetime. The
154        // deallocation of the node happens outside of this block, after the lifetime
155        // ends.
156        let inner_node = unsafe { inner_ptr.as_ref() };
157
158        // SAFETY: This iterator only lives for this block, a subset of the shared
159        // lifetime of the `inner_node` variable. By the safety requirements of the
160        // `deallocate_inner_node` function, no other mutation of this node can happen
161        // while this iterator is live.
162        let iter = inner_node.iter();
163        stack.extend(iter.map(|(_, child)| child).filter(filter_children));
164    }
165
166    // SAFETY: The single call per node requirement is enforced by the safety
167    // requirements on this function.
168    drop(unsafe { NodePtr::deallocate_node_ptr(inner_ptr, alloc) });
169}
