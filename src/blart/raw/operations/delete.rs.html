13/// Remove a child node from the given inner node, return the child node
14/// pointer if it was compressed.
15///
16/// The inner node will be compressed if there was only a single child
17/// remaining after the delete. Compressing the node involves prepending the
18/// inner node key prefix and child key byte to the child's key prefix.
19///
20/// # Safety
21///  - `inner_node_ptr` must be a unique pointer to the node and must not have
22///    any other mutable references.
23///  - There must not be any mutable references to the children of the given
24///    inner node either.
25///  - `alloc` must be the same allocator that was used to allocate the nodes of
26///    the trie.
27unsafe fn remove_child_from_inner_node_and_compress<
28    const PREFIX_LEN: usize,
29    N: InnerNode<PREFIX_LEN>,
30    A: Allocator,
31>(
32    inner_node_ptr: NodePtr<PREFIX_LEN, N>,
33    key_fragment: u8,
34    #[cfg(debug_assertions)] expected_leaf_node: NodePtr<
35        PREFIX_LEN,
36        LeafNode<N::Key, N::Value, PREFIX_LEN>,
37    >,
38    alloc: &A,
39) -> Option<OpaqueNodePtr<N::Key, N::Value, PREFIX_LEN>> {
40    // SAFETY: The `inner_node` reference is scoped to this function and dropped
41    // before cases where the inner node is deallocated. It is a unique reference,
42    // by the safety requirements of the containing function.
43    let inner_node = unsafe { inner_node_ptr.as_mut() };
44
45    let _removed = inner_node.remove_child(key_fragment);
46
47    #[cfg(debug_assertions)]
48    {
49        debug_assert!(
50            _removed.is_some(),
51            "child must be present at [{key_fragment}] in inner node [{inner_node:?}]"
52        );
53        let removed = _removed.unwrap();
54        debug_assert!(
55            removed.is::<LeafNode<N::Key, N::Value, PREFIX_LEN>>(),
56            "child at [{key_fragment}] in inner node [{inner_node:?}] must be a leaf node"
57        );
58        debug_assert_eq!(
59            removed,
60            expected_leaf_node.to_opaque(),
61            "child at [{key_fragment}] in inner node [{inner_node:?}] should be [{:?}]",
62            expected_leaf_node.to_opaque()
63        );
64    }
65
66    if inner_node.header().num_children() == 1 {
67        // need to compress node into child
68
69        // SAFETY: The iterator only lasts until the remaining child is pulled out, then
70        // dropped. The scope does not overlap with any mutating operations on the inner
71        // node.
72        let mut children = inner_node.iter();
73        let (child_key_byte, child_node_ptr) = children.next().expect("expected single child");
74        debug_assert!(
75            children.next().is_none(),
76            "expected only a single child, not more"
77        );
78        drop(children);
79
80        // SAFETY: By the safety requirements of the function, there are no other
81        // references to this child node. The reference only lasts for the scope of this
82        // `if` block.
83        if let Some(child_header) = unsafe { child_node_ptr.header_mut() } {
84            // Construct the new prefix, by concatenating the parent header, child_key_byte,
85            // child header.
86            //
87            // Here we can use the fact that the header of both the child and the parent can
88            // hold up to PREFIX_LEN, so if for example the parent header len >=
89            // PREFIX_LEN, the new child header will hold only this bytes since
90            // the size is already greater than the capacity.
91            //
92            // so we can "clear" the child prefix, by setting the len to 0 and repopulate by
93            // pushing the parent header, child_key_byte, child prefix. If the
94            // header if already full we don't copy, just increment the len by
95            // the size.
96            let parent_header = inner_node.header();
97            let parent_prefix = parent_header.read_prefix();
98            let parent_len = parent_header.prefix_len();
99
100            let (old_prefix, old_len, old_capped_len) = child_header.clear_prefix();
101            child_header.push_prefix(parent_prefix, parent_len);
102            child_header.push_prefix(&[child_key_byte], 1);
103            child_header.push_prefix(&old_prefix[..old_capped_len], old_len);
104        }
105        // the else case here is that the child does not have a header, and
106        // is a leaf
107
108        // SAFETY: Since this function requires a unique pointer to the original
109        // `inner_node_ptr`, we know that no other code will deallocate the pointer
110        unsafe {
111            drop(NodePtr::deallocate_node_ptr(inner_node_ptr, alloc));
112        }
113
114        Some(child_node_ptr)
115    } else if N::TYPE.should_shrink_inner_node(inner_node.header().num_children()) {
116        let new_inner_node = inner_node.shrink();
117
118        let new_inner_node_ptr = NodePtr::allocate_node_ptr(new_inner_node, alloc).to_opaque();
119
120        // SAFETY: Since this function requires a unique pointer to the original
121        // `inner_node_ptr`, we know that no other code will deallocate the pointer
122        unsafe {
123            drop(NodePtr::deallocate_node_ptr(inner_node_ptr, alloc));
124        }
125
126        Some(new_inner_node_ptr)
127    } else {
128        None
129    }
130}

132/// Delete the given non-root leaf node.
133///
134/// # Safety
135///  - `parent_node_ptr` must be a unique pointer to the node and must not have
136///    any other mutable references.
137///  - There must not be any other mutable references to any children of the
138///    `parent_node_ptr` either.
139///  - `grandparent_node_ptr` must be a unique pointer to the node and must not
140///    have any other mutable references.
141///  - `leaf_node_ptr` must be a unique pointer to the node and not have any
142///    other mutable references.
143///  - `alloc` must be the same allocator that was used to allocate the nodes of
144///    the trie.
145unsafe fn inner_delete_non_root_unchecked<K, V, const PREFIX_LEN: usize, A: Allocator>(
146    leaf_node_ptr: NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>,
147    parent_node_ptr: OpaqueNodePtr<K, V, PREFIX_LEN>,
148    child_key_byte: u8,
149    grandparent_node_ptr: Option<(OpaqueNodePtr<K, V, PREFIX_LEN>, u8)>,
150    original_root: OpaqueNodePtr<K, V, PREFIX_LEN>,
151    alloc: &A,
152) -> DeleteResult<K, V, PREFIX_LEN> {
153    let new_parent_node_ptr = match_concrete_node_ptr!(match (parent_node_ptr.to_node_ptr()) {
154        InnerNode(parent_node_ptr) => unsafe {
155            // SAFETY: Covered by containing function safety doc
156            remove_child_from_inner_node_and_compress(
157                parent_node_ptr,
158                child_key_byte,
159                #[cfg(debug_assertions)]
160                leaf_node_ptr,
161                alloc,
162            )
163        },
164        LeafNode(_leaf) => unreachable!("Cannot have delete from leaf node"),
165    });
166
167    // If the parent node was changed to something else, we have to write the new
168    // value to the grandparent
169    if let Some(new_parent_node_ptr) = new_parent_node_ptr {
170        if let Some((grandparent_node_ptr, parent_key_byte)) = grandparent_node_ptr {
171            match_concrete_node_ptr!(match (grandparent_node_ptr.to_node_ptr()) {
172                InnerNode(inner_node_ptr) => {
173                    // SAFETY: The scope of the mutable reference is limited to this block, and
174                    // the containing function safety requirements mean that there are no other
175                    // mutable references to the same node.
176                    let inner_node = unsafe { inner_node_ptr.as_mut() };
177                    inner_node.write_child(parent_key_byte, new_parent_node_ptr);
178                },
179                LeafNode(_leaf) => {
180                    unreachable!("Cannot modify children of a leaf node")
181                },
182            })
183        }
184    }
185
186    // SAFETY: Function safety doc covers the no concurrent read or modification of
187    // this leaf node.
188    unsafe { LeafNode::remove_self(leaf_node_ptr) }
189
190    // SAFETY: `leaf_node_ptr` is a unique pointer to the leaf node, no other code
191    // will deallocate this
192    let leaf_node = unsafe { NodePtr::deallocate_node_ptr(leaf_node_ptr, alloc) };
193
194    let new_root = match (new_parent_node_ptr, grandparent_node_ptr) {
195        (Some(new_parent_node_ptr), None) => new_parent_node_ptr,
196        _ => original_root,
197    };
198
199    DeleteResult {
200        new_root: Some(new_root),
201        deleted_leaf: leaf_node,
202    }
203}

205/// The results of a successful delete operation
206#[derive(Debug)]
207pub struct DeleteResult<K, V, const PREFIX_LEN: usize> {
208    /// The new root node for the tree, after the delete has been applied.
209    ///
210    /// If `None`, that means the tree is now empty.
211    pub new_root: Option<OpaqueNodePtr<K, V, PREFIX_LEN>>,
212
213    /// The leaf node that was successfully deleted.
214    pub deleted_leaf: LeafNode<K, V, PREFIX_LEN>,
215}

217/// This struct represents a location in the trie that can be deleted.
218pub struct DeletePoint<K, V, const PREFIX_LEN: usize> {
219    /// The path to the leaf node to delete.
220    pub path: TreePath<K, V, PREFIX_LEN>,
221
222    /// The node to delete.
223    pub leaf_node_ptr: NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>,
224}

226impl<K, V, const PREFIX_LEN: usize> fmt::Debug for DeletePoint<K, V, PREFIX_LEN> {
227    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
228        f.debug_struct("DeletePoint")
229            .field("path", &self.path)
230            .field("leaf_node_ptr", &self.leaf_node_ptr)
231            .finish()
232    }
233}

235impl<K, V, const PREFIX_LEN: usize> DeletePoint<K, V, PREFIX_LEN> {
236    /// Handle the logic of deleting a leaf node from the tree, after it has
237    /// been found.
238    ///
239    /// # Safety
240    ///  - The `root` [`OpaqueNodePtr`] must be a unique pointer to the
241    ///    underlying tree
242    ///  - This function cannot be called concurrently to any reads or writes of
243    ///    the `root` node or any child node of `root`. This function will
244    ///    arbitrarily read or write to any child in the given tree.
245    ///  - `alloc` must be the same allocator that was used to allocate the
246    ///    nodes of the trie.
247    ///  - This function may invalidate existing pointers into the trie when
248    ///    leaves are deleted and when inner nodes are deleted or shrunk.
249    ///    Callers must ensure that they delete invalidated pointers, the new
250    ///    pointers are returned in [`DeleteResult`].
251    pub unsafe fn apply<A: Allocator>(
252        self,
253        root: OpaqueNodePtr<K, V, PREFIX_LEN>,
254        alloc: &A,
255    ) -> DeleteResult<K, V, PREFIX_LEN> {
256        match self.path {
257            TreePath::Root => {
258                // The leaf node was also the root node. We don't need to remove the leaf from
259                // the linked list here because the `previous` and `next` pointers should both
260                // be `None`, since it is the only leaf.
261
262                // SAFETY: The original `root` node pointer is a unique pointer to the tree
263                // (required by safety doc), which means that leaf_node_ptr is also unique and
264                // can be deallocated.
265                let leaf_node = unsafe { NodePtr::deallocate_node_ptr(self.leaf_node_ptr, alloc) };
266
267                DeleteResult {
268                    new_root: None,
269                    deleted_leaf: leaf_node,
270                }
271            },
272            TreePath::ChildOfRoot {
273                parent: root_parent,
274                child_key_byte,
275            } => unsafe {
276                // SAFETY: `root` is a unique pointer to the tree and there will be no
277                // concurrent reads or writes to any portion of the tree, so all these child
278                // nodes will be unique pointers and not read/written.
279                inner_delete_non_root_unchecked(
280                    self.leaf_node_ptr,
281                    root_parent,
282                    child_key_byte,
283                    None,
284                    root,
285                    alloc,
286                )
287            },
288            TreePath::Normal {
289                grandparent,
290                parent_key_byte,
291                parent,
292                child_key_byte,
293            } => unsafe {
294                // SAFETY: `root` is a unique pointer to the tree and there will be no
295                // concurrent reads or writes to any portion of the tree, so all these child
296                // nodes will be unique pointers and not read/written.
297                inner_delete_non_root_unchecked(
298                    self.leaf_node_ptr,
299                    parent,
300                    child_key_byte,
301                    Some((grandparent, parent_key_byte)),
302                    root,
303                    alloc,
304                )
305            },
306        }
307    }
308}

310/// Search in the given tree for the leaf node to delete, returning `None` if it
311/// does not exist.
312///
313/// This function also returns information that will be required in the delete
314/// process, like the parent and possibly grandparent nodes.
315///
316/// # Safety
317///  - This function cannot be called concurrently with any mutating operation
318///    on `root` or any child node of `root`. This function will arbitrarily
319///    read to any child in the given tree.
320pub unsafe fn search_for_delete_point<K, V, const PREFIX_LEN: usize>(
321    root: OpaqueNodePtr<K, V, PREFIX_LEN>,
322    key_bytes: &[u8],
323) -> Option<DeletePoint<K, V, PREFIX_LEN>>
324where
325    K: AsBytes,
326{
327    let mut path = TreePathSearch::default();
328    let mut current_node = root;
329    let mut current_depth = 0;
330    let mut prefix_match_state = PrefixMatchBehavior::default();
331
332    loop {
333        let next_node = match_concrete_node_ptr!(match (current_node.to_node_ptr()) {
334            InnerNode(inner_ptr) => unsafe {
335                // SAFETY: The safety requirement is covered by the safety requirement on the
336                // containing function
337                lookup::check_prefix_lookup_child(
338                    inner_ptr,
339                    key_bytes,
340                    &mut current_depth,
341                    &mut prefix_match_state,
342                )
343            },
344            LeafNode(leaf_node_ptr) => {
345                // SAFETY: The created reference doesn't escape the block and there are no
346                // concurrent mutable references.
347                let leaf = unsafe { leaf_node_ptr.as_ref() };
348
349                // Specifically we are matching the leaf node stored key against the full search
350                // key to confirm that it is the right value.
351                if prefix_match_state.matches_leaf_key(leaf, key_bytes, current_depth) {
352                    return Some(DeletePoint {
353                        path: path.finish(),
354                        leaf_node_ptr,
355                    });
356                } else {
357                    return None;
358                }
359            },
360        })?;
361
362        debug_assert!(
363            current_depth > 0,
364            "for a non-leaf node, there should be some amount of key"
365        );
366
367        // This should not panic because the current_depth will be greater than zero
368        let last_key_byte = key_bytes[current_depth - 1];
369
370        path.visit_inner_node(current_node, last_key_byte);
371        current_node = next_node;
372    }
373}

375/// Find the minimum leaf in the tree and return the information necessary to
376/// delete it.
377///
378/// # Safety
379///  - This function cannot be called concurrently with any mutating operation
380///    on `root` or any child node of `root`. This function will arbitrarily
381///    read to any child in the given tree.
382#[inline]
383pub unsafe fn find_minimum_to_delete<K, V, const PREFIX_LEN: usize>(
384    root: OpaqueNodePtr<K, V, PREFIX_LEN>,
385) -> DeletePoint<K, V, PREFIX_LEN> {
386    let mut path = TreePathSearch::default();
387    let mut current_node = root;
388
389    loop {
390        // SAFETY: We hold a mutable reference, so creating
391        // a shared reference is safe
392        let (last_key_byte, next_node) =
393            match_concrete_node_ptr!(match (current_node.to_node_ptr()) {
394                InnerNode(inner_node) => unsafe { inner_node.as_ref().min() },
395                LeafNode(leaf_node_ptr) => {
396                    return DeletePoint {
397                        path: path.finish(),
398                        leaf_node_ptr,
399                    };
400                },
401            });
402
403        path.visit_inner_node(current_node, last_key_byte);
404        current_node = next_node;
405    }
406}
