10/// The common header for all inner nodes
11#[derive(Clone, PartialEq, Eq)]
12pub struct Header<const PREFIX_LEN: usize> {
13    /// Number of children of this inner node. This field has no meaning for
14    /// a leaf node.
15    ///
16    /// This needs to be a [`u16`], since a node 256 can hold up to 256 children
17    /// if this was a [`u8`] (0-255) it would overflow when adding the last
18    /// element
19    num_children: u16,
20    /// Number of bytes used by the prefix
21    prefix_len: u32,
22    /// The key prefix for this node.
23    prefix: [u8; PREFIX_LEN],
24}
25
26impl<const PREFIX_LEN: usize> Header<PREFIX_LEN> {
27    /// Create a new header with the given prefix and prefix length.
28    ///
29    /// The given prefix length must be greater than or equal to `prefix.len()`.
30    /// If it is longer, it represents the case where there are implicit bytes
31    /// in the prefix.
32    ///
33    /// There can also be implicit bytes in the prefix in the case that
34    /// `prefix.len()` > `PREFIX_LEN`, since the header will only store
35    /// `PREFIX_LEN` bytes directly.
36    #[inline]
37    pub fn new(prefix: &[u8], prefix_len: usize) -> Self {
38        debug_assert!(
39            prefix_len >= prefix.len(),
40            "Explicit prefix length must be greater than or equal to the length of the actual \
41             prefix passed."
42        );
43
44        let mut header = Self {
45            num_children: 0,
46            prefix_len: prefix_len as u32,
47            prefix: [0; PREFIX_LEN],
48        };
49        let len = prefix.len().min(PREFIX_LEN);
50        header.prefix[..len].copy_from_slice(&prefix[..len]);
51
52        header
53    }
54
55    /// Create a new `Header` for an empty node.
56    #[inline]
57    pub fn empty() -> Self {
58        Self {
59            num_children: 0,
60            prefix_len: 0,
61            prefix: [0; PREFIX_LEN],
62        }
63    }
64
65    /// Read the initialized portion of the prefix present in the header.
66    #[inline]
67    pub fn read_prefix(&self) -> &[u8] {
68        &self.prefix[0..self.capped_prefix_len()]
69    }
70
71    /// Get the number of bytes in the prefix.
72    #[inline]
73    pub fn prefix_len(&self) -> usize {
74        self.prefix_len as usize
75    }
76
77    /// Minimum between [`Self::prefix_len`] and `PREFIX_LEN`.
78    #[inline]
79    pub fn capped_prefix_len(&self) -> usize {
80        (self.prefix_len as usize).min(PREFIX_LEN)
81    }
82
83    /// Return the number of children of this node.
84    #[inline]
85    pub fn num_children(&self) -> usize {
86        usize::from(self.num_children)
87    }
88
89    /// Left trim by `len`, copies the remaining data to the beginning of the
90    /// prefix
91    ///
92    /// # Panics
93    ///  - If `len` > length of the prefix
94    #[inline]
95    pub fn ltrim_by(&mut self, len: usize) {
96        assert!(
97            (len as u32) <= self.prefix_len,
98            "given length [{len}] must be less than or equal to the prefix length [{}]",
99            self.prefix_len
100        );
101        self.prefix_len -= len as u32;
102
103        let begin = len;
104        let end = begin + self.capped_prefix_len();
105
106        unsafe {
107            // SAFETY: This function is called when mismatch happened and
108            // we used the node to match the number of bytes,
109            // by this we know that len < prefix len, but since we + 1,
110            // to skip the key byte we have that len <= prefix len
111            core::hint::assert_unchecked(end <= self.prefix.len());
112
113            // SAFETY: This is by construction end = begin + len
114            core::hint::assert_unchecked(begin <= end);
115        }
116        self.prefix.copy_within(begin..end, 0);
117    }
118
119    /// Set the length of the prefix to 0 and returns a copy of the
120    /// prefix, length and capped length
121    #[inline]
122    pub fn clear_prefix(&mut self) -> ([u8; PREFIX_LEN], usize, usize) {
123        let len = self.prefix_len();
124        let capped_len = self.capped_prefix_len();
125        self.prefix_len = 0;
126
127        (self.prefix, len, capped_len)
128    }
129
130    /// Append `new` to the prefix and sums `new_len` to the prefix length
131    #[inline]
132    pub fn push_prefix(&mut self, new: &[u8], new_len: usize) {
133        let begin = self.capped_prefix_len();
134        let end = (begin + new.len()).min(PREFIX_LEN);
135        let len = end - begin;
136        self.prefix[begin..end].copy_from_slice(&new[..len]);
137        self.prefix_len += new_len as u32;
138    }
139
140    /// Increments the number of children
141    #[inline]
142    pub fn inc_num_children(&mut self) {
143        self.num_children += 1;
144    }
145
146    /// Decrements the number of children
147    #[inline]
148    pub fn dec_num_children(&mut self) {
149        self.num_children -= 1;
150    }
151
152    /// Reset the number of children to 0.
153    #[inline]
154    pub fn reset_num_children(&mut self) {
155        self.num_children = 0;
156    }
157
158    /// Remove the `len` length starting portion of this header's prefix and
159    /// refill the prefix value with contents from the given leaf.
160    ///
161    /// # Safety
162    ///
163    /// This function must not be called concurrently with any other read or
164    /// modification of the given leaf or header.
165    #[inline]
166    pub unsafe fn ltrim_by_with_leaf<K: AsBytes, V>(
167        &mut self,
168        len: usize,
169        depth: usize,
170        leaf_ptr: NodePtr<PREFIX_LEN, LeafNode<K, V, PREFIX_LEN>>,
171    ) {
172        self.prefix_len -= len as u32;
173
174        // SAFETY: Since have a mutable reference
175        // is safe to create a shared reference from it
176        let leaf_key = unsafe { leaf_ptr.as_key_ref().as_bytes() };
177
178        let begin = depth + len;
179        let end = begin + self.capped_prefix_len();
180        let len = end - begin;
181
182        unsafe {
183            // SAFETY: This function is called a mismatch happened and
184            // we used the leaf to match the number of matching bytes,
185            // by this we know that len < prefix len, but since we + 1,
186            // to skip the key byte we have that len <= prefix len
187            core::hint::assert_unchecked(end <= leaf_key.len());
188
189            // SAFETY: This is by construction end = begin + len
190            core::hint::assert_unchecked(begin <= end);
191        }
192
193        let leaf_key = &leaf_key[begin..end];
194        self.prefix[..len].copy_from_slice(leaf_key)
195    }
196}
197
198impl<const PREFIX_LEN: usize> Debug for Header<PREFIX_LEN> {
199    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
200        f.debug_struct("Header")
201            .field("num_children", &self.num_children)
202            .field("prefix_len", &self.prefix_len)
203            .field(
204                "prefix",
205                &&self.prefix[..(self.prefix_len as usize).min(self.prefix.len())],
206            )
207            .finish()
208    }
209}

211#[cfg(test)]
212mod tests {
213    use super::*;
214
215    #[test]
216    fn header_delete_prefix() {
217        let mut h = Header::<16>::new(&[1, 2, 3, 4, 5, 6, 7, 8], 8);
218        assert_eq!(h.read_prefix(), &[1, 2, 3, 4, 5, 6, 7, 8]);
219        assert_eq!(h.prefix_len(), 8);
220
221        h.ltrim_by(0);
222        assert_eq!(h.read_prefix(), &[1, 2, 3, 4, 5, 6, 7, 8]);
223        assert_eq!(h.prefix_len(), 8);
224
225        h.ltrim_by(3);
226        assert_eq!(h.read_prefix(), &[4, 5, 6, 7, 8]);
227        assert_eq!(h.prefix_len(), 5);
228
229        h.ltrim_by(1);
230        assert_eq!(h.read_prefix(), &[5, 6, 7, 8]);
231        assert_eq!(h.prefix_len(), 4);
232
233        h.ltrim_by(4);
234        assert_eq!(h.read_prefix(), &[] as &[u8]);
235        assert_eq!(h.prefix_len(), 0);
236    }

238    #[test]
239    #[should_panic = "given length [10] must be less than or equal to the prefix length [8]"]
240    fn header_ltrim_prefix_too_many_bytes_panic() {
241        let mut h = Header::<16>::new(&[1, 2, 3, 4, 5, 6, 7, 8], 8);
242        assert_eq!(h.read_prefix(), &[1, 2, 3, 4, 5, 6, 7, 8]);
243        assert_eq!(h.prefix_len(), 8);
244
245        h.ltrim_by(10);
246    }
