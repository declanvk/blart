24/// A restricted index only valid from 0 to LIMIT - 1.
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26#[repr(transparent)]
27pub struct RestrictedNodeIndex<const LIMIT: u8>(u8);
28
29impl<const LIMIT: u8> RestrictedNodeIndex<LIMIT> {
30    /// A placeholder index value that indicates that the index is not
31    /// occupied
32    pub const EMPTY: Self = RestrictedNodeIndex(LIMIT);
33
34    /// Return true if the given index is the empty sentinel value
35    pub fn is_empty(self) -> bool {
36        self == Self::EMPTY
37    }
38}

40impl<const LIMIT: u8> From<RestrictedNodeIndex<LIMIT>> for u8 {
41    fn from(src: RestrictedNodeIndex<LIMIT>) -> Self {
42        src.0
43    }
44}

46impl<const LIMIT: u8> From<RestrictedNodeIndex<LIMIT>> for usize {
47    fn from(src: RestrictedNodeIndex<LIMIT>) -> Self {
48        usize::from(src.0)
49    }
50}

52impl<const LIMIT: u8> TryFrom<usize> for RestrictedNodeIndex<LIMIT> {
53    type Error = TryFromByteError;
54
55    fn try_from(value: usize) -> Result<Self, Self::Error> {
56        if value < usize::from(LIMIT) {
57            Ok(RestrictedNodeIndex(value as u8))
58        } else {
59            Err(TryFromByteError(LIMIT, value))
60        }
61    }
62}

64impl<const LIMIT: u8> TryFrom<u8> for RestrictedNodeIndex<LIMIT> {
65    type Error = TryFromByteError;
66
67    fn try_from(value: u8) -> Result<Self, Self::Error> {
68        if value < LIMIT {
69            Ok(RestrictedNodeIndex(value))
70        } else {
71            Err(TryFromByteError(LIMIT, usize::from(value)))
72        }
73    }

76impl<const LIMIT: u8> PartialOrd for RestrictedNodeIndex<LIMIT> {
77    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
78        if self.0 == LIMIT || other.0 == LIMIT {
79            None
80        } else {
81            Some(self.0.cmp(&other.0))
82        }
83    }

140impl<K, V, const PREFIX_LEN: usize> InnerNode48<K, V, PREFIX_LEN> {
141    /// Return the initialized portions of the child pointer array.
142    pub fn initialized_child_pointers(&self) -> &[OpaqueNodePtr<K, V, PREFIX_LEN>] {
143        unsafe {
144            // SAFETY: The array prefix with length `header.num_children` is guaranteed to
145            // be initialized
146            core::hint::assert_unchecked(self.header.num_children() <= self.child_pointers.len());
147            maybe_uninit_slice_assume_init_ref(&self.child_pointers[..self.header.num_children()])

159impl<K, V, const PREFIX_LEN: usize> InnerNode<PREFIX_LEN> for InnerNode48<K, V, PREFIX_LEN> {
160    type GrownNode = InnerNode256<K, V, PREFIX_LEN>;
161    #[cfg(not(feature = "nightly"))]
162    type Iter<'a>
163        = Node48Iter<'a, K, V, PREFIX_LEN>
164    where
165        Self: 'a;
166    #[cfg(feature = "nightly")]
167    type Iter<'a>
168        = Map<
169        FilterMap<
170            Enumerate<Iter<'a, RestrictedNodeIndex<48>>>,
171            impl FnMut((usize, &'a RestrictedNodeIndex<48>)) -> Option<(u8, usize)>,
172        >,
173        impl FnMut((u8, usize)) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>),
174    >
175    where
176        Self: 'a;
177    type ShrunkNode = InnerNode16<K, V, PREFIX_LEN>;
178
179    fn header(&self) -> &Header<PREFIX_LEN> {
180        &self.header
181    }
182
183    fn from_header(header: Header<PREFIX_LEN>) -> Self {
184        InnerNode48 {
185            header,
186            child_indices: [RestrictedNodeIndex::<48>::EMPTY; 256],
187            child_pointers: [MaybeUninit::uninit(); 48],
188        }
189    }
190
191    fn lookup_child(&self, key_fragment: u8) -> Option<OpaqueNodePtr<K, V, PREFIX_LEN>> {
192        let index = &self.child_indices[usize::from(key_fragment)];
193        let child_pointers = self.initialized_child_pointers();
194        if !index.is_empty() {
195            let idx = usize::from(*index);
196
197            unsafe {
198                // SAFETY: If `idx` is out of bounds we have more than
199                // 48 children in this node, so it should have already
200                // grown. So it's safe to assume that it's in bounds
201                core::hint::assert_unchecked(idx < child_pointers.len());
202            }
203            Some(child_pointers[idx])
204        } else {
205            None
206        }
207    }
208
209    fn write_child(&mut self, key_fragment: u8, child_pointer: OpaqueNodePtr<K, V, PREFIX_LEN>) {
210        let key_fragment_idx = usize::from(key_fragment);
211        let child_index = if self.child_indices[key_fragment_idx] == RestrictedNodeIndex::EMPTY {
212            let child_index = self.header.num_children();
213            debug_assert!(child_index < self.child_pointers.len(), "node is full");
214
215            // SAFETY: By construction the number of children in the header
216            // is kept in sync with the number of children written in the node
217            // and if this number exceeds the maximum len the node should have
218            // already grown. So we know for a fact that that num_children <= node len.
219            //
220            // With this we know that child_index is <= 47, because at the 48th time
221            // calling this function for writing, the current len will bet 47, and
222            // after this insert we increment it to 48, so this symbolizes that the
223            // node is full and before calling this function again the node should
224            // have already grown
225            self.child_indices[key_fragment_idx] =
226                unsafe { RestrictedNodeIndex::try_from(child_index).unwrap_unchecked() };
227            self.header.inc_num_children();
228            child_index
229        } else {
230            // overwrite existing
231            usize::from(self.child_indices[key_fragment_idx])
232        };
233
234        // SAFETY: This index can be up to <= 47 as described above
235
236        unsafe {
237            core::hint::assert_unchecked(child_index < self.child_pointers.len());
238            self.child_pointers
239                .get_unchecked_mut(child_index)
240                .write(child_pointer);
241        }
242    }
243
244    fn remove_child(&mut self, key_fragment: u8) -> Option<OpaqueNodePtr<K, V, PREFIX_LEN>> {
245        let restricted_index = self.child_indices[usize::from(key_fragment)];
246        if restricted_index.is_empty() {
247            return None;
248        }
249
250        // Replace child pointer with uninitialized value, even though it may possibly
251        // be overwritten by the compaction step
252        let child_ptr = mem::replace(
253            &mut self.child_pointers[usize::from(restricted_index)],
254            MaybeUninit::uninit(),
255        );
256
257        // Copy all the child_pointer values in higher indices down by one.
258        self.child_pointers.copy_within(
259            (usize::from(restricted_index) + 1)..self.header.num_children(),
260            usize::from(restricted_index),
261        );
262
263        // Take all child indices that are greater than the index we're removing, and
264        // subtract one so that they remain valid
265        for other_restrict_index in &mut self.child_indices {
266            if matches!(
267                restricted_index.partial_cmp(other_restrict_index),
268                Some(Ordering::Less)
269            ) {
270                // PANIC SAFETY: This will not underflow, because it is guaranteed to be
271                // greater than at least 1 other index. This will not panic in the
272                // `try_from` because the new value is derived from an existing restricted
273                // index.
274                *other_restrict_index =
275                    RestrictedNodeIndex::try_from(other_restrict_index.0 - 1).unwrap();
276            }
277        }
278
279        self.child_indices[usize::from(key_fragment)] = RestrictedNodeIndex::EMPTY;
280        self.header.dec_num_children();
281        // SAFETY: This child pointer value is initialized because we got it by using a
282        // non-`RestrictedNodeIndex::<>::EMPTY` index from the child indices array.
283        Some(unsafe { MaybeUninit::assume_init(child_ptr) })
284    }
285
286    fn grow(&self) -> Self::GrownNode {
287        let header = self.header.clone();
288        let mut child_pointers = [None; 256];
289        let initialized_child_pointers = self.initialized_child_pointers();
290        for (key_fragment, idx) in self.child_indices.iter().enumerate() {
291            if idx.is_empty() {
292                continue;
293            }
294
295            let idx = usize::from(*idx);
296
297            unsafe {
298                // SAFETY: When growing initialized_child_pointers should be full
299                // i.e initialized_child_pointers len == 48. And idx <= 47, since
300                // we can't insert in a full, node
301                core::hint::assert_unchecked(idx < initialized_child_pointers.len());
302            }
303            let child_pointer = initialized_child_pointers[idx];
304            child_pointers[key_fragment] = Some(child_pointer);
305        }
306
307        InnerNode256 {
308            header,
309            child_pointers,
310        }
311    }
312
313    fn shrink(&self) -> Self::ShrunkNode {
314        assert!(
315            self.header.num_children() <= 16,
316            "Cannot shrink a Node48 when it has more than 16 children. Currently has [{}] \
317             children.",
318            self.header.num_children()
319        );
320
321        let header = self.header.clone();
322
323        let mut key_and_child_ptrs = [MaybeUninit::uninit(); 16];
324
325        for (idx, value) in self.iter().enumerate() {
326            key_and_child_ptrs[idx].write(value);
327        }
328
329        let init_key_and_child_ptrs = {
330            // SAFETY: The first `num_children` are guaranteed to be initialized in this
331            // array because the previous iterator loops through all children of the inner
332            // node.
333            let init_key_and_child_ptrs = unsafe {
334                maybe_uninit_slice_assume_init_mut(&mut key_and_child_ptrs[..header.num_children()])
335            };
336
337            init_key_and_child_ptrs.sort_unstable_by_key(|(key_byte, _)| *key_byte);
338
339            init_key_and_child_ptrs
340        };
341
342        let mut keys = [MaybeUninit::uninit(); 16];
343        let mut child_pointers = [MaybeUninit::uninit(); 16];
344
345        for (idx, (key_byte, child_ptr)) in init_key_and_child_ptrs.iter().copied().enumerate() {
346            keys[idx].write(key_byte);
347            child_pointers[idx].write(child_ptr);
348        }
349
350        InnerNodeCompressed {
351            header,
352            keys,
353            child_pointers,
354        }
355    }

357    fn iter(&self) -> Self::Iter<'_> {
358        let child_pointers = self.initialized_child_pointers();
359
360        #[cfg(not(feature = "nightly"))]
361        {
362            Node48Iter {
363                it: self.child_indices.iter().enumerate(),
364                child_pointers,
365            }
366        }
367
368        #[cfg(feature = "nightly")]
369        {
370            self.child_indices
371                .iter()
372                .enumerate()
373                .filter_map(|(key, idx)| {
374                    (!idx.is_empty()).then_some((key as u8, usize::from(*idx)))
375                })
376                // SAFETY: By the construction of `Self` idx, must always
377                // be inbounds
378                .map(|(key, idx)| unsafe { (key, *child_pointers.get_unchecked(idx)) })
379        }
380    }

382    fn range(
383        &self,
384        bound: impl core::ops::RangeBounds<u8>,
385    ) -> impl DoubleEndedIterator<Item = (u8, OpaqueNodePtr<Self::Key, Self::Value, PREFIX_LEN>)>
386           + FusedIterator {
387        {
388            match (bound.start_bound(), bound.end_bound()) {
389                (Bound::Excluded(s), Bound::Excluded(e)) if s == e => {
390                    panic!("range start and end are equal and excluded: ({s:?})")
391                },
392                (
393                    Bound::Included(s) | Bound::Excluded(s),
394                    Bound::Included(e) | Bound::Excluded(e),
395                ) if s > e => {
396                    panic!("range start ({s:?}) is greater than range end ({e:?})")
397                },
398                _ => {},
399            }
400        }
401
402        let child_pointers = self.initialized_child_pointers();
403
404        let start = bound.start_bound().map(|val| usize::from(*val));
405        let key_offset = match bound.start_bound() {
406            core::ops::Bound::Included(val) => *val,
407            core::ops::Bound::Excluded(val) => val.saturating_add(1),
408            core::ops::Bound::Unbounded => 0,
409        };
410        let end = bound.end_bound().map(|val| usize::from(*val));
411
412        self.child_indices[(start, end)]
413            .iter()
414            .enumerate()
415            .filter_map(|(key, idx)| {
416                // normally `enumerate()` has elements of (idx, val), but we're using the index
417                // as the key since it ranges from [0, 256)
418                (!idx.is_empty()).then_some((key as u8, usize::from(*idx)))
419            })
420            // SAFETY: By the construction of `Self` idx, must always
421            // be inbounds
422            .map(move |(key, idx)| unsafe {
423                (
424                    key.saturating_add(key_offset),
425                    *child_pointers.get_unchecked(idx),
426                )
427            })
428    }

430    #[cfg(feature = "nightly")]
431    #[cfg_attr(test, mutants::skip)]
432    fn min(&self) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>) {
433        // SAFETY: Since `RestrictedNodeIndex` is
434        // repr(u8) is safe to transmute it
435        let child_indices: &[u8; 256] = unsafe { core::mem::transmute(&self.child_indices) };
436        let empty = u8x64::splat(48);
437        let r0 = u8x64::from_array(child_indices[0..64].try_into().unwrap())
438            .simd_eq(empty)
439            .to_bitmask();
440        let r1 = u8x64::from_array(child_indices[64..128].try_into().unwrap())
441            .simd_eq(empty)
442            .to_bitmask();
443        let r2 = u8x64::from_array(child_indices[128..192].try_into().unwrap())
444            .simd_eq(empty)
445            .to_bitmask();
446        let r3 = u8x64::from_array(child_indices[192..256].try_into().unwrap())
447            .simd_eq(empty)
448            .to_bitmask();
449
450        let key = if r0 != u64::MAX {
451            r0.trailing_ones()
452        } else if r1 != u64::MAX {
453            r1.trailing_ones() + 64
454        } else if r2 != u64::MAX {
455            r2.trailing_ones() + 128
456        } else {
457            r3.trailing_ones() + 192
458        } as usize;
459
460        unsafe {
461            // SAFETY: key can be at up to 256, but we are in a inner node
462            // this means that this node has at least 1 child (it's even more
463            // strict since, if we have 1 child the node would collapse), so we
464            // know that exists at least one idx where != 48
465            core::hint::assert_unchecked(key < self.child_indices.len());
466        }
467
468        let idx = usize::from(self.child_indices[key]);
469        let child_pointers = self.initialized_child_pointers();
470
471        unsafe {
472            // SAFETY: We know that idx is in bounds, because the value can't be
473            // constructed if it >= 48 and also it has to be < num children, since
474            // it's constructed from the num children before being incremented during
475            // insertion process
476            core::hint::assert_unchecked(idx < child_pointers.len());
477        }
478
479        (key as u8, child_pointers[idx])
480    }

495    #[cfg_attr(test, mutants::skip)]
496    fn max(&self) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>) {
497        // SAFETY: Since `RestrictedNodeIndex` is
498        // repr(u8) is safe to transmute it
499        let child_indices: &[u8; 256] = unsafe { core::mem::transmute(&self.child_indices) };
500        let empty = u8x64::splat(48);
501        let r0 = u8x64::from_array(child_indices[0..64].try_into().unwrap())
502            .simd_eq(empty)
503            .to_bitmask();
504        let r1 = u8x64::from_array(child_indices[64..128].try_into().unwrap())
505            .simd_eq(empty)
506            .to_bitmask();
507        let r2 = u8x64::from_array(child_indices[128..192].try_into().unwrap())
508            .simd_eq(empty)
509            .to_bitmask();
510        let r3 = u8x64::from_array(child_indices[192..256].try_into().unwrap())
511            .simd_eq(empty)
512            .to_bitmask();
513
514        let key = if r3 != u64::MAX {
515            255 - r3.leading_ones()
516        } else if r2 != u64::MAX {
517            191 - r2.leading_ones()
518        } else if r1 != u64::MAX {
519            127 - r1.leading_ones()
520        } else {
521            // SAFETY: This subtraction can't fail, because we know that
522            // we have at least one child, so the number of leading ones
523            // in this last case is <= 63
524            63 - r0.leading_ones()
525        } as usize;
526
527        unsafe {
528            // SAFETY: idx can be at up to 255 so it's in bounds
529            core::hint::assert_unchecked(key < self.child_indices.len());
530        }
531
532        let idx = usize::from(self.child_indices[key]);
533        let child_pointers = self.initialized_child_pointers();
534
535        unsafe {
536            // SAFETY: We know that idx is in bounds, because the value can't be
537            // constructed if it >= 48 and also it has to be < num children, since
538            // it's constructed from the num children before being incremented during
539            // insertion process
540            core::hint::assert_unchecked(idx < child_pointers.len());
541        }
542
543        (key as u8, child_pointers[idx])
544    }

546    #[cfg(not(feature = "nightly"))]
547    fn max(&self) -> (u8, OpaqueNodePtr<K, V, PREFIX_LEN>) {
548        for (key, idx) in self.child_indices.iter().enumerate().rev() {
549            if idx.is_empty() {

551            }

554        }

556    }

559/// This struct is an iterator over the children of a [`InnerNode48`].
560#[cfg(not(feature = "nightly"))]
561pub struct Node48Iter<'a, K, V, const PREFIX_LEN: usize> {
562    pub(crate) it: Enumerate<Iter<'a, RestrictedNodeIndex<48>>>,
563    pub(crate) child_pointers: &'a [OpaqueNodePtr<K, V, PREFIX_LEN>],
564}
565
566#[cfg(not(feature = "nightly"))]
567impl<K, V, const PREFIX_LEN: usize> Iterator for Node48Iter<'_, K, V, PREFIX_LEN> {
568    type Item = (u8, OpaqueNodePtr<K, V, PREFIX_LEN>);
569
570    fn next(&mut self) -> Option<Self::Item> {
571        for (key, idx) in self.it.by_ref() {
572            if idx.is_empty() {
573                continue;
574            }
575            let key = key as u8;
576            // SAFETY: This idx is in bounds, since the number
577            // of iterations is always <= 48 (i.e 0-47)
578            let child_pointer = unsafe { *self.child_pointers.get_unchecked(usize::from(*idx)) };
579            return Some((key, child_pointer));
580        }
581        None
582    }
583}

586impl<K, V, const PREFIX_LEN: usize> DoubleEndedIterator for Node48Iter<'_, K, V, PREFIX_LEN> {
587    fn next_back(&mut self) -> Option<Self::Item> {

596            return Some((key, child_pointer));
597        }
598        None
599    }
